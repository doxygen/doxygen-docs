42/** Helper class representing the stack of items considered while resolving
43 * the scope.
44 */
45class AccessStack
46{
47 /** Element in the stack. */
48 struct AccessElem
49 {
50 AccessElem(const Definition *d,const FileDef *f,const Definition *i) : scope(d), fileScope(f), item(i) {}
51 AccessElem(const Definition *d,const FileDef *f,const Definition *i,const QCString &e) : scope(d), fileScope(f), item(i), expScope(e) {}
52 const Definition *scope;
53 const FileDef *fileScope;
54 const Definition *item;
55 QCString expScope;
56 };
57 public:
58 void push(const Definition *scope,const FileDef *fileScope,const Definition *item)
59 {
60 m_elements.emplace_back(scope,fileScope,item);
61 }
62 void push(const Definition *scope,const FileDef *fileScope,const Definition *item,const QCString &expScope)
63 {
64 m_elements.emplace_back(scope,fileScope,item,expScope);
65 }
66 void pop()
67 {
68 if (!m_elements.empty()) m_elements.pop_back();
69 }
70 bool find(const Definition *scope,const FileDef *fileScope, const Definition *item)
71 {
72 auto it = std::find_if(m_elements.begin(),m_elements.end(),
73 [&](const AccessElem &e) { return e.scope==scope && e.fileScope==fileScope && e.item==item; });
74 return it!=m_elements.end();
75 }
76 bool find(const Definition *scope,const FileDef *fileScope, const Definition *item,const QCString &expScope)
77 {
78 auto it = std::find_if(m_elements.begin(),m_elements.end(),
79 [&](const AccessElem &e) { return e.scope==scope && e.fileScope==fileScope && e.item==item && e.expScope==expScope; });
80 return it!=m_elements.end();
81 }
82 void clear()
83 {
84 m_elements.clear();
85 }
86
87 private:
88 std::vector<AccessElem> m_elements;
89};

215const ClassDef *SymbolResolver::Private::getResolvedTypeRec(
216 VisitedKeys &visitedKeys,
217 const Definition *scope,
218 const QCString &n,
219 const MemberDef **pTypeDef,
220 QCString *pTemplSpec,
221 QCString *pResolvedType)
222{
223 AUTO_TRACE("scope={} name={}",scope->name(),n);
224 if (n.isEmpty()) return nullptr;
225 QCString explicitScopePart;
226 QCString strippedTemplateParams;
227 QCString scopeName=scope!=Doxygen::globalScope ? scope->name() : QCString();
228 QCString name=stripTemplateSpecifiersFromScope(n,TRUE,&strippedTemplateParams,scopeName);
229 std::unique_ptr<ArgumentList> actTemplParams;
230 if (!strippedTemplateParams.isEmpty()) // template part that was stripped
231 {
232 actTemplParams = stringToArgumentList(scope->getLanguage(),strippedTemplateParams);
233 }
234
235 int qualifierIndex = computeQualifiedIndex(name);
236 //printf("name=%s qualifierIndex=%d\n",qPrint(name),qualifierIndex);
237 if (qualifierIndex!=-1) // qualified name
238 {
239 // split off the explicit scope part
240 explicitScopePart=name.left(qualifierIndex);
241 // todo: improve namespace alias substitution
242 replaceNamespaceAliases(explicitScopePart);
243 name=name.mid(qualifierIndex+2);
244 }
245
246 if (name.isEmpty())
247 {
248 AUTO_TRACE_EXIT("empty name");
249 return nullptr; // empty name
250 }
251
252 auto &range = Doxygen::symbolMap->find(name);
253 if (range.empty())
254 {
255 AUTO_TRACE_EXIT("no symbol with this name");
256 return nullptr;
257 }
258
259 bool hasUsingStatements =
260 (m_fileScope && (!m_fileScope->getUsedNamespaces().empty() ||
261 !m_fileScope->getUsedDefinitions().empty())
262 );
263 // Since it is often the case that the same name is searched in the same
264 // scope over an over again (especially for the linked source code generation)
265 // we use a cache to collect previous results. This is possible since the
266 // result of a lookup is deterministic. As the key we use the concatenated
267 // scope, the name to search for and the explicit scope prefix. The speedup
268 // achieved by this simple cache can be enormous.
269 size_t scopeNameLen = scope->name().length()+1;
270 size_t nameLen = name.length()+1;
271 size_t explicitPartLen = explicitScopePart.length();
272 size_t fileScopeLen = hasUsingStatements ? 1+m_fileScope->absFilePath().length() : 0;
273
274 // below is a more efficient coding of
275 // QCString key=scope->name()+"+"+name+"+"+explicitScopePart+args+typesOnly?'T':'F';
276 QCString key(scopeNameLen+nameLen+explicitPartLen+fileScopeLen, QCString::ExplicitSize);
277 char *pk=key.rawData();
278 qstrcpy(pk,scope->name().data()); *(pk+scopeNameLen-1)='+';
279 pk+=scopeNameLen;
280 qstrcpy(pk,name.data()); *(pk+nameLen-1)='+';
281 pk+=nameLen;
282 qstrcpy(pk,explicitScopePart.data());
283 pk+=explicitPartLen;
284
285 // if a file scope is given and it contains using statements we should
286 // also use the file part in the key (as a class name can be in
287 // two different namespaces and a using statement in a file can select
288 // one of them).
289 if (hasUsingStatements)
290 {
291 // below is a more efficient coding of
292 // key+="+"+m_fileScope->name();
293 *pk++='+';
294 qstrcpy(pk,m_fileScope->absFilePath().data());
295 pk+=fileScopeLen-1;
296 }
297 *pk='\0';
298
299 const ClassDef *bestMatch=nullptr;
300 {
301 if (std::find(visitedKeys.begin(),visitedKeys.end(),key.str())!=std::end(visitedKeys))
302 {
303 // we are already in the middle of find the definition for this key.
304 // avoid recursion
305 AUTO_TRACE_EXIT("recursion detected");
306 return nullptr;
307 }
308 // remember the key
309 visitedKeys.push_back(key.str());
310
311 LookupInfo *pval = nullptr;
312 {
313 std::lock_guard lock(g_cacheMutex);
314 pval = Doxygen::typeLookupCache->find(key.str());
315 }
316 AUTO_TRACE_ADD("key={} found={}",key,pval!=nullptr);
317 if (pval)
318 {
319 if (pTemplSpec) *pTemplSpec=pval->templSpec;
320 if (pTypeDef) *pTypeDef=pval->typeDef;
321 if (pResolvedType) *pResolvedType=pval->resolvedType;
322 AUTO_TRACE_EXIT("found cached name={} templSpec={} typeDef={} resolvedTypedef={}",
323 pval->definition?pval->definition->name():QCString(),
324 pval->templSpec,
325 pval->typeDef?pval->typeDef->name():QCString(),
326 pval->resolvedType);
327
328 return toClassDef(pval->definition);
329 }
330
331 const MemberDef *bestTypedef=nullptr;
332 QCString bestTemplSpec;
333 QCString bestResolvedType;
334 int minDistance=10000; // init at "infinite"
335
336 for (Definition *d : range)
337 {
338 getResolvedType(visitedKeys,scope,d,explicitScopePart,actTemplParams.get(),
339 minDistance,bestMatch,bestTypedef,bestTemplSpec,bestResolvedType);
340 if (minDistance==0) break; // we can stop reaching if we already reached distance 0
341 }
342
343 if (pTypeDef)
344 {
345 *pTypeDef = bestTypedef;
346 }
347 if (pTemplSpec)
348 {
349 *pTemplSpec = bestTemplSpec;
350 }
351 if (pResolvedType)
352 {
353 *pResolvedType = bestResolvedType;
354 }
355
356 {
357 std::lock_guard lock(g_cacheMutex);
358 Doxygen::typeLookupCache->insert(key.str(),
359 LookupInfo(bestMatch,bestTypedef,bestTemplSpec,bestResolvedType));
360 }
361 visitedKeys.erase(std::remove(visitedKeys.begin(), visitedKeys.end(), key.str()), visitedKeys.end());
362
363 AUTO_TRACE_EXIT("found name={} templSpec={} typeDef={} resolvedTypedef={}",
364 bestMatch?bestMatch->name():QCString(),
365 bestTemplSpec,
366 bestTypedef?bestTypedef->name():QCString(),
367 bestResolvedType);
368 }
369 return bestMatch;
370}

372const Definition *SymbolResolver::Private::getResolvedSymbolRec(
373 VisitedKeys &visitedKeys,
374 const Definition *scope,
375 const QCString &n,
376 const QCString &args,
377 bool checkCV,
378 bool insideCode,
379 const MemberDef **pTypeDef,
380 QCString *pTemplSpec,
381 QCString *pResolvedType)
382{
383 AUTO_TRACE("scope={} name={} args={} checkCV={} insideCode={}",
384 scope->name(),n,args,checkCV,insideCode);
385 if (n.isEmpty()) return nullptr;
386 QCString explicitScopePart;
387 QCString strippedTemplateParams;
388 QCString scopeName=scope!=Doxygen::globalScope ? scope->name() : QCString();
389 QCString name=stripTemplateSpecifiersFromScope(n,TRUE,&strippedTemplateParams,scopeName);
390 std::unique_ptr<ArgumentList> actTemplParams;
391 if (!strippedTemplateParams.isEmpty()) // template part that was stripped
392 {
393 actTemplParams = stringToArgumentList(scope->getLanguage(),strippedTemplateParams);
394 }
395
396 int qualifierIndex = computeQualifiedIndex(name);
397 //printf("name=%s qualifierIndex=%d\n",qPrint(name),qualifierIndex);
398 if (qualifierIndex!=-1) // qualified name
399 {
400 // split off the explicit scope part
401 explicitScopePart=name.left(qualifierIndex);
402 // todo: improve namespace alias substitution
403 replaceNamespaceAliases(explicitScopePart);
404 name=name.mid(qualifierIndex+2);
405 }
406 AUTO_TRACE_ADD("qualifierIndex={} name={} explicitScopePart={}",qualifierIndex,name,explicitScopePart);
407
408 if (name.isEmpty())
409 {
410 AUTO_TRACE_EXIT("empty name qualifierIndex={}",qualifierIndex);
411 return nullptr; // empty name
412 }
413
414 int i=0;
415 const auto &range1 = Doxygen::symbolMap->find(name);
416 const auto &range = (range1.empty() && (i=name.find('<'))!=-1) ? Doxygen::symbolMap->find(name.left(i)) : range1;
417 if (range.empty())
418 {
419 AUTO_TRACE_ADD("no symbols with name '{}' (including unspecialized)",name);
420 return nullptr;
421 }
422 AUTO_TRACE_ADD("{} -> {} candidates",name,range.size());
423
424 bool hasUsingStatements =
425 (m_fileScope && (!m_fileScope->getUsedNamespaces().empty() ||
426 !m_fileScope->getUsedDefinitions().empty())
427 );
428 // Since it is often the case that the same name is searched in the same
429 // scope over an over again (especially for the linked source code generation)
430 // we use a cache to collect previous results. This is possible since the
431 // result of a lookup is deterministic. As the key we use the concatenated
432 // scope, the name to search for and the explicit scope prefix. The speedup
433 // achieved by this simple cache can be enormous.
434 size_t scopeNameLen = scope!=Doxygen::globalScope ? scope->name().length()+1 : 0;
435 size_t nameLen = name.length()+1;
436 size_t explicitPartLen = explicitScopePart.length();
437 size_t fileScopeLen = hasUsingStatements ? 1+m_fileScope->absFilePath().length() : 0;
438 size_t argsLen = args.length()+1;
439
440 // below is a more efficient coding of
441 // QCString key=scope->name()+"+"+name+"+"+explicitScopePart+args+typesOnly?'T':'F';
442 std::string key;
443 key.reserve(scopeNameLen+nameLen+explicitPartLen+fileScopeLen+argsLen);
444 if (scope!=Doxygen::globalScope)
445 {
446 key+=scope->name().str();
447 key+='+';
448 }
449 key+=name.str();
450 key+='+';
451 key+=explicitScopePart.str();
452
453 // if a file scope is given and it contains using statements we should
454 // also use the file part in the key (as a class name can be in
455 // two different namespaces and a using statement in a file can select
456 // one of them).
457 if (hasUsingStatements)
458 {
459 // below is a more efficient coding of
460 // key+="+"+m_fileScope->name();
461 key+='+';
462 key+=m_fileScope->absFilePath().str();
463 }
464 if (argsLen>0)
465 {
466 key+='+';
467 key+=args.str();
468 }
469
470 const Definition *bestMatch=nullptr;
471 {
472 if (std::find(visitedKeys.begin(),visitedKeys.end(),key)!=std::end(visitedKeys))
473 {
474 // we are already in the middle of find the definition for this key.
475 // avoid recursion
476 return nullptr;
477 }
478 // remember the key
479 visitedKeys.push_back(key);
480 LookupInfo *pval = nullptr;
481 {
482 std::lock_guard lock(g_cacheMutex);
483 pval = Doxygen::symbolLookupCache->find(key);
484 }
485 AUTO_TRACE_ADD("key={} found={}",key,pval!=nullptr);
486 if (pval)
487 {
488 if (pTemplSpec) *pTemplSpec=pval->templSpec;
489 if (pTypeDef) *pTypeDef=pval->typeDef;
490 if (pResolvedType) *pResolvedType=pval->resolvedType;
491 AUTO_TRACE_EXIT("found cached name={} templSpec={} typeDef={} resolvedTypedef={}",
492 pval->definition?pval->definition->name():QCString(),
493 pval->templSpec,
494 pval->typeDef?pval->typeDef->name():QCString(),
495 pval->resolvedType);
496 return pval->definition;
497 }
498
499 const MemberDef *bestTypedef=nullptr;
500 QCString bestTemplSpec;
501 QCString bestResolvedType;
502 int minDistance=10000; // init at "infinite"
503
504 for (Definition *d : range)
505 {
506 getResolvedSymbol(visitedKeys,scope,d,args,checkCV,insideCode,explicitScopePart,false,
507 minDistance,bestMatch,bestTypedef,bestTemplSpec,bestResolvedType);
508 if (minDistance==0) break; // we can stop reaching if we already reached distance 0
509 }
510
511 // in case we are looking for e.g. func() and the real function is func(int x) we also
512 // accept func(), see example 036 in the test set.
513 if (bestMatch==nullptr && args=="()")
514 {
515 for (Definition *d : range)
516 {
517 getResolvedSymbol(visitedKeys,scope,d,QCString(),false,insideCode,explicitScopePart,true,
518 minDistance,bestMatch,bestTypedef,bestTemplSpec,bestResolvedType);
519 if (minDistance==0) break; // we can stop reaching if we already reached distance 0
520 }
521 }
522
523 if (pTypeDef)
524 {
525 *pTypeDef = bestTypedef;
526 }
527 if (pTemplSpec)
528 {
529 *pTemplSpec = bestTemplSpec;
530 }
531 if (pResolvedType)
532 {
533 *pResolvedType = bestResolvedType;
534 }
535
536 {
537 LookupInfo lookupInfo(bestMatch,bestTypedef,bestTemplSpec,bestResolvedType);
538 std::lock_guard lock(g_cacheMutex);
539 // we need to insert the item in the cache again, as it could be removed in the meantime
540 Doxygen::symbolLookupCache->insert(key,std::move(lookupInfo));
541 }
542 visitedKeys.erase(std::remove(visitedKeys.begin(),visitedKeys.end(),key),visitedKeys.end());
543
544 AUTO_TRACE_EXIT("found name={} templSpec={} typeDef={} resolvedTypedef={}",
545 bestMatch?bestMatch->name():QCString(),
546 bestTemplSpec,
547 bestTypedef?bestTypedef->name():QCString(),
548 bestResolvedType);
549 }
550 return bestMatch;
551}

553void SymbolResolver::Private::getResolvedType(
554 VisitedKeys &visitedKeys, // in
555 const Definition *scope, // in
556 const Definition *d, // in
557 const QCString &explicitScopePart, // in
558 const ArgumentList *actTemplParams, // in
559 int &minDistance, // inout
560 const ClassDef *&bestMatch, // out
561 const MemberDef *&bestTypedef, // out
562 QCString &bestTemplSpec, // out
563 QCString &bestResolvedType // out
564 )
565{
566 AUTO_TRACE("scope={} sym={} explicitScope={}",scope->name(),d->qualifiedName(),explicitScopePart);
567 // only look at classes and members that are enums or typedefs
568 if (d->definitionType()==Definition::TypeClass ||
569 (d->definitionType()==Definition::TypeMember &&
570 ((toMemberDef(d))->isTypedef() ||
571 (toMemberDef(d))->isEnumerate())
572 )
573 )
574 {
575 VisitedNamespaces visitedNamespaces;
576 AccessStack accessStack;
577 // test accessibility of definition within scope.
578 int distance = isAccessibleFromWithExpScope(visitedKeys,visitedNamespaces,
579 accessStack,scope,d,explicitScopePart);
580 AUTO_TRACE_ADD("distance={}",distance);
581 if (distance!=-1) // definition is accessible
582 {
583 // see if we are dealing with a class or a typedef
584 if (d->definitionType()==Definition::TypeClass) // d is a class
585 {
586 const ClassDef *cd = toClassDef(d);
587 //printf("cd=%s\n",qPrint(cd->name()));
588 if (!cd->isTemplateArgument()) // skip classes that
589 // are only there to
590 // represent a template
591 // argument
592 {
593 //printf("is not a templ arg\n");
594 if (distance<minDistance) // found a definition that is "closer"
595 {
596 AUTO_TRACE_ADD("found symbol={} at distance={} minDistance={}",cd->name(),distance,minDistance);
597 minDistance=distance;
598 bestMatch = cd;
599 bestTypedef = nullptr;
600 bestTemplSpec.clear();
601 bestResolvedType = cd->qualifiedName();
602 }
603 else if (distance==minDistance &&
604 m_fileScope && bestMatch &&
605 !m_fileScope->getUsedNamespaces().empty() &&
606 d->getOuterScope()->definitionType()==Definition::TypeNamespace &&
607 bestMatch->getOuterScope()==Doxygen::globalScope
608 )
609 {
610 // in case the distance is equal it could be that a class X
611 // is defined in a namespace and in the global scope. When searched
612 // in the global scope the distance is 0 in both cases. We have
613 // to choose one of the definitions: we choose the one in the
614 // namespace if the fileScope imports namespaces and the definition
615 // found was in a namespace while the best match so far isn't.
616 // Just a non-perfect heuristic but it could help in some situations
617 // (kdecore code is an example).
618 AUTO_TRACE_ADD("found symbol={} at distance={} minDistance={}",cd->name(),distance,minDistance);
619 minDistance=distance;
620 bestMatch = cd;
621 bestTypedef = nullptr;
622 bestTemplSpec.clear();
623 bestResolvedType = cd->qualifiedName();
624 }
625 }
626 else
627 {
628 //printf(" is a template argument!\n");
629 }
630 }
631 else if (d->definitionType()==Definition::TypeMember)
632 {
633 const MemberDef *md = toMemberDef(d);
634 AUTO_TRACE_ADD("member={} isTypeDef={}",md->name(),md->isTypedef());
635 if (md->isTypedef()) // d is a typedef
636 {
637 QCString args=md->argsString();
638 if (args.isEmpty()) // do not expand "typedef t a[4];"
639 {
640 // we found a symbol at this distance, but if it didn't
641 // resolve to a class, we still have to make sure that
642 // something at a greater distance does not match, since
643 // that symbol is hidden by this one.
644 if (distance<minDistance)
645 {
646 QCString spec;
647 QCString type;
648 minDistance=distance;
649 const MemberDef *enumType = nullptr;
650 const ClassDef *cd = newResolveTypedef(visitedKeys,scope,md,&enumType,&spec,&type,actTemplParams);
651 if (cd) // type resolves to a class
652 {
653 AUTO_TRACE_ADD("found symbol={} at distance={} minDistance={}",cd->name(),distance,minDistance);
654 bestMatch = cd;
655 bestTypedef = md;
656 bestTemplSpec = spec;
657 bestResolvedType = type;
658 }
659 else if (enumType) // type resolves to a member type
660 {
661 AUTO_TRACE_ADD("found enum");
662 bestMatch = nullptr;
663 bestTypedef = enumType;
664 bestTemplSpec = "";
665 bestResolvedType = enumType->qualifiedName();
666 }
667 else if (md->isReference()) // external reference
668 {
669 AUTO_TRACE_ADD("found external reference");
670 bestMatch = nullptr;
671 bestTypedef = md;
672 bestTemplSpec = spec;
673 bestResolvedType = type;
674 }
675 else
676 {
677 AUTO_TRACE_ADD("no match");
678 bestMatch = nullptr;
679 bestTypedef = md;
680 bestTemplSpec.clear();
681 bestResolvedType.clear();
682 }
683 }
684 else
685 {
686 //printf(" not the best match %d min=%d\n",distance,minDistance);
687 }
688 }
689 else
690 {
691 AUTO_TRACE_ADD("skipping complex typedef");
692 }
693 }
694 else if (md->isEnumerate())
695 {
696 if (distance<minDistance)
697 {
698 AUTO_TRACE_ADD("found enum={} at distance={} minDistance={}",md->name(),distance,minDistance);
699 minDistance=distance;
700 bestMatch = nullptr;
701 bestTypedef = md;
702 bestTemplSpec = "";
703 bestResolvedType = md->qualifiedName();
704 }
705 }
706 }
707 } // if definition accessible
708 else
709 {
710 AUTO_TRACE_ADD("not accessible");
711 }
712 } // if definition is a class or member
713 AUTO_TRACE_EXIT("bestMatch sym={} type={}",
714 bestMatch?bestMatch->name():QCString("<none>"),bestResolvedType);
715}

717
718void SymbolResolver::Private::getResolvedSymbol(
719 VisitedKeys &visitedKeys, // in
720 const Definition *scope, // in
721 const Definition *d, // in
722 const QCString &args, // in
723 bool checkCV, // in
724 bool insideCode, // in
725 const QCString &explicitScopePart, // in
726 bool forceCallable, // in
727 int &minDistance, // inout
728 const Definition *&bestMatch, // out
729 const MemberDef *&bestTypedef, // out
730 QCString &bestTemplSpec, // out
731 QCString &bestResolvedType // out
732 )
733{
734 AUTO_TRACE("scope={} sym={}",scope->name(),d->qualifiedName());
735 // only look at classes and members that are enums or typedefs
736 VisitedNamespaces visitedNamespaces;
737 AccessStack accessStack;
738 // test accessibility of definition within scope.
739 int distance = isAccessibleFromWithExpScope(visitedKeys,visitedNamespaces,accessStack,scope,d,explicitScopePart);
740 AUTO_TRACE_ADD("distance={}",distance);
741 if (distance!=-1) // definition is accessible
742 {
743 // see if we are dealing with a class or a typedef
744 if (args.isEmpty() && !forceCallable && d->definitionType()==Definition::TypeClass) // d is a class
745 {
746 const ClassDef *cd = toClassDef(d);
747 if (!cd->isTemplateArgument()) // skip classes that
748 // are only there to
749 // represent a template
750 // argument
751 {
752 if (distance<minDistance) // found a definition that is "closer"
753 {
754 AUTO_TRACE_ADD("found symbol={} at distance={} minDistance={}",d->name(),distance,minDistance);
755 minDistance=distance;
756 bestMatch = d;
757 bestTypedef = nullptr;
758 bestTemplSpec.clear();
759 bestResolvedType = cd->qualifiedName();
760 }
761 else if (distance==minDistance &&
762 m_fileScope && bestMatch &&
763 !m_fileScope->getUsedNamespaces().empty() &&
764 d->getOuterScope()->definitionType()==Definition::TypeNamespace &&
765 bestMatch->getOuterScope()==Doxygen::globalScope
766 )
767 {
768 // in case the distance is equal it could be that a class X
769 // is defined in a namespace and in the global scope. When searched
770 // in the global scope the distance is 0 in both cases. We have
771 // to choose one of the definitions: we choose the one in the
772 // namespace if the fileScope imports namespaces and the definition
773 // found was in a namespace while the best match so far isn't.
774 // Just a non-perfect heuristic but it could help in some situations
775 // (kdecore code is an example).
776 AUTO_TRACE_ADD("found symbol={} at distance={} minDistance={}",d->name(),distance,minDistance);
777 minDistance=distance;
778 bestMatch = d;
779 bestTypedef = nullptr;
780 bestTemplSpec.clear();
781 bestResolvedType = cd->qualifiedName();
782 }
783 }
784 else
785 {
786 AUTO_TRACE_ADD("class with template arguments");
787 }
788 }
789 else if (d->definitionType()==Definition::TypeMember)
790 {
791 const MemberDef *md = toMemberDef(d);
792
793 bool match = true;
794 AUTO_TRACE_ADD("member={} args={} isCallable()={}",md->name(),argListToString(md->argumentList()),md->isCallable());
795 if (md->isCallable() && !args.isEmpty())
796 {
797 QCString actArgs;
798 if (md->isArtificial() && md->formalTemplateArguments()) // for members of an instantiated template we need to replace
799 // the formal arguments by the actual ones before matching
800 // See issue #10640
801 {
802 actArgs = substituteTemplateArgumentsInString(args,md->formalTemplateArguments().value(),&md->argumentList());
803 }
804 else
805 {
806 actArgs = args;
807 }
808 std::unique_ptr<ArgumentList> argList = stringToArgumentList(md->getLanguage(),actArgs);
809 const ArgumentList &mdAl = md->argumentList();
810 match = matchArguments2(md->getOuterScope(),md->getFileDef(),&mdAl,
811 scope, md->getFileDef(),argList.get(),
812 checkCV,md->getLanguage());
813 AUTO_TRACE_ADD("match={}",match);
814 }
815
816 if (match && distance<minDistance)
817 {
818 AUTO_TRACE_ADD("found symbol={} at distance={} minDistance={}",md->name(),distance,minDistance);
819 minDistance=distance;
820 bestMatch = md;
821 bestTypedef = md;
822 bestTemplSpec = "";
823 bestResolvedType = md->qualifiedName();
824 }
825 }
826 else if ((d->definitionType()==Definition::TypeNamespace ||
827 d->definitionType()==Definition::TypeFile))
828 {
829 if (distance<minDistance) // found a definition that is "closer"
830 {
831 AUTO_TRACE_ADD("found symbol={} at distance={} minDistance={}",d->name(),distance,minDistance);
832 minDistance=distance;
833 bestMatch = d;
834 bestTypedef = nullptr;
835 bestTemplSpec.clear();
836 bestResolvedType.clear();
837 }
838 }
839 } // if definition accessible
840 else
841 {
842 AUTO_TRACE_ADD("not accessible");
843 }
844 AUTO_TRACE_EXIT("bestMatch sym={} distance={}",
845 bestMatch?bestMatch->name():QCString("<none>"),bestResolvedType);
846}

849const ClassDef *SymbolResolver::Private::newResolveTypedef(
850 VisitedKeys &visitedKeys, // in
851 const Definition * /* scope */, // in
852 const MemberDef *md, // in
853 const MemberDef **pMemType, // out
854 QCString *pTemplSpec, // out
855 QCString *pResolvedType, // out
856 const ArgumentList *actTemplParams) // in
857{
858 AUTO_TRACE("md={}",md->qualifiedName());
859 std::lock_guard<std::recursive_mutex> lock(g_cacheTypedefMutex);
860 bool isCached = md->isTypedefValCached(); // value already cached
861 if (isCached)
862 {
863 AUTO_TRACE_EXIT("cached typedef={} resolvedTypedef={} templSpec={}",
864 md->getCachedTypedefVal() ? md->getCachedTypedefVal()->name() : QCString(),
865 md->getCachedResolvedTypedef(),
866 md->getCachedTypedefTemplSpec());
867
868 if (pTemplSpec) *pTemplSpec = md->getCachedTypedefTemplSpec();
869 if (pResolvedType) *pResolvedType = md->getCachedResolvedTypedef();
870 return md->getCachedTypedefVal();
871 }
872
873 QCString qname = md->qualifiedName();
874 if (m_resolvedTypedefs.find(qname.str())!=m_resolvedTypedefs.end())
875 {
876 AUTO_TRACE_EXIT("already being processed");
877 return nullptr; // typedef already done
878 }
879
880 auto typedef_it = m_resolvedTypedefs.emplace(qname.str(),md).first; // put on the trace list
881
882 const ClassDef *typeClass = md->getClassDef();
883 QCString type = md->typeString(); // get the "value" of the typedef
884 if (typeClass && typeClass->isTemplate() &&
885 actTemplParams && !actTemplParams->empty())
886 {
887 type = substituteTemplateArgumentsInString(type,
888 typeClass->templateArguments(),actTemplParams);
889 }
890 QCString typedefValue = type;
891 int tl=static_cast<int>(type.length());
892 int ip=tl-1; // remove * and & at the end
893 while (ip>=0 && (type.at(ip)=='*' || type.at(ip)=='&' || type.at(ip)==' '))
894 {
895 ip--;
896 }
897 type=type.left(ip+1);
898 type.stripPrefix("const "); // strip leading "const"
899 type.stripPrefix("volatile "); // strip leading "volatile"
900 type.stripPrefix("struct "); // strip leading "struct"
901 type.stripPrefix("union "); // strip leading "union"
902 int sp=0;
903 tl=static_cast<int>(type.length()); // length may have been changed
904 while (sp<tl && type.at(sp)==' ') sp++;
905 const MemberDef *memTypeDef = nullptr;
906 const ClassDef *result = getResolvedTypeRec(visitedKeys,md->getOuterScope(),type,
907 &memTypeDef,nullptr,pResolvedType);
908 // if type is a typedef then return what it resolves to.
909 if (memTypeDef && memTypeDef->isTypedef())
910 {
911 AUTO_TRACE_ADD("resolving typedef");
912 result=newResolveTypedef(visitedKeys,m_fileScope,memTypeDef,pMemType,pTemplSpec,nullptr);
913 goto done;
914 }
915 else if (memTypeDef && memTypeDef->isEnumerate() && pMemType)
916 {
917 *pMemType = memTypeDef;
918 }
919
920 if (result==nullptr)
921 {
922 // try unspecialized version if type is template
923 int si=type.findRev("::");
924 int i=type.find('<');
925 if (si==-1 && i!=-1) // typedef of a template => try the unspecialized version
926 {
927 if (pTemplSpec) *pTemplSpec = type.mid(i);
928 result = getResolvedTypeRec(visitedKeys,md->getOuterScope(),type.left(i),nullptr,nullptr,pResolvedType);
929 }
930 else if (si!=-1) // A::B
931 {
932 i=type.find('<',si);
933 if (i==-1) // Something like A<T>::B => lookup A::B
934 {
935 i=static_cast<int>(type.length());
936 }
937 else // Something like A<T>::B<S> => lookup A::B, spec=<S>
938 {
939 if (pTemplSpec) *pTemplSpec = type.mid(i);
940 }
941 result = getResolvedTypeRec(visitedKeys,md->getOuterScope(),
942 stripTemplateSpecifiersFromScope(type.left(i),FALSE),nullptr,nullptr,pResolvedType);
943 }
944 }
945
946done:
947 if (pResolvedType)
948 {
949 if (result && result->definitionType()==Definition::TypeClass)
950 {
951 *pResolvedType = result->qualifiedName();
952 if (sp>0) pResolvedType->prepend(typedefValue.left(sp));
953 if (ip<tl-1) pResolvedType->append(typedefValue.right(tl-ip-1));
954 }
955 else
956 {
957 *pResolvedType = typedefValue;
958 }
959 }
960
961 // remember computed value for next time
962 if (result && result->getDefFileName()!="<code>")
963 // this check is needed to prevent that temporary classes that are
964 // introduced while parsing code fragments are being cached here.
965 {
966 AUTO_TRACE_ADD("caching typedef relation {}->{}",md->name(),result->name());
967 MemberDefMutable *mdm = toMemberDefMutable(const_cast<MemberDef*>(md));
968 if (mdm)
969 {
970 mdm->cacheTypedefVal(result,
971 pTemplSpec ? *pTemplSpec : QCString(),
972 pResolvedType ? *pResolvedType : QCString()
973 );
974 }
975 }
976
977 m_resolvedTypedefs.erase(typedef_it); // remove from the trace list
978
979 AUTO_TRACE_EXIT("result={} pTemplSpec={} pResolvedType={}",
980 result ? result->name() : QCString(),
981 pTemplSpec ? *pTemplSpec : "<nullptr>",
982 pResolvedType ? *pResolvedType : "<nullptr>"
983 );
984 return result;
985}

996int SymbolResolver::Private::isAccessibleFromWithExpScope(
997 VisitedKeys &visitedKeys,
998 VisitedNamespaces &visitedNamespaces,
999 AccessStack &accessStack,
1000 const Definition *scope,
1001 const Definition *item,
1002 const QCString &explicitScopePart)
1003{
1004 int result=0; // assume we found it
1005 AUTO_TRACE("scope={} item={} explictScopePart={}",
1006 scope?scope->name():QCString(), item?item->name():QCString(), explicitScopePart);
1007 if (explicitScopePart.isEmpty())
1008 {
1009 // handle degenerate case where there is no explicit scope.
1010 result = isAccessibleFrom(visitedKeys,accessStack,scope,item);
1011 AUTO_TRACE_EXIT("result={}",result);
1012 return result;
1013 }
1014
1015 if (accessStack.find(scope,m_fileScope,item,explicitScopePart))
1016 {
1017 AUTO_TRACE_EXIT("already found");
1018 return -1;
1019 }
1020 accessStack.push(scope,m_fileScope,item,explicitScopePart);
1021
1022 const Definition *newScope = followPath(visitedKeys,scope,explicitScopePart);
1023 if (newScope) // explicitScope is inside scope => newScope is the result
1024 {
1025 Definition *itemScope = item->getOuterScope();
1026
1027 AUTO_TRACE_ADD("scope traversal successful newScope={}",newScope->name());
1028
1029 bool nestedClassInsideBaseClass =
1030 itemScope &&
1031 itemScope->definitionType()==Definition::TypeClass &&
1032 newScope->definitionType()==Definition::TypeClass &&
1033 (toClassDef(newScope))->isBaseClass(toClassDef(itemScope),TRUE);
1034
1035 bool enumValueWithinEnum =
1036 item->definitionType()==Definition::TypeMember &&
1037 toMemberDef(item)->isEnumValue() &&
1038 toMemberDef(item)->getEnumScope()==newScope;
1039
1040 if (itemScope==newScope) // exact match of scopes => distance==0
1041 {
1042 AUTO_TRACE_ADD("found scope match");
1043 }
1044 else if (nestedClassInsideBaseClass)
1045 {
1046 // inheritance is also ok. Example: looking for B::I, where
1047 // class A { public: class I {} };
1048 // class B : public A {}
1049 // but looking for B::I, where
1050 // class A { public: class I {} };
1051 // class B { public: class I {} };
1052 // will find A::I, so we still prefer a direct match and give this one a distance of 1
1053 result=1;
1054
1055 AUTO_TRACE_ADD("{} is a bass class of {}",scope->name(),newScope->name());
1056 }
1057 else if (enumValueWithinEnum)
1058 {
1059 AUTO_TRACE_ADD("found enum value inside enum");
1060 result=1;
1061 }
1062 else
1063 {
1064 int i=-1;
1065 if (newScope->definitionType()==Definition::TypeNamespace)
1066 {
1067 visitedNamespaces.emplace(newScope->name().str(),newScope);
1068 // this part deals with the case where item is a class
1069 // A::B::C but is explicit referenced as A::C, where B is imported
1070 // in A via a using directive.
1071 //printf("newScope is a namespace: %s!\n",qPrint(newScope->name()));
1072 const NamespaceDef *nscope = toNamespaceDef(newScope);
1073 for (const auto &ud : nscope->getUsedDefinitions())
1074 {
1075 if (ud==item)
1076 {
1077 AUTO_TRACE_ADD("found in used definition {}",ud->name());
1078 goto done;
1079 }
1080 }
1081 for (const auto &nd : nscope->getUsedNamespaces())
1082 {
1083 if (visitedNamespaces.find(nd->name().str())==visitedNamespaces.end())
1084 {
1085 i = isAccessibleFromWithExpScope(visitedKeys,visitedNamespaces,accessStack,scope,item,nd->name());
1086 if (i!=-1)
1087 {
1088 AUTO_TRACE_ADD("found in used namespace {}",nd->name());
1089 goto done;
1090 }
1091 }
1092 }
1093 }
1094#if 0 // this caused problems resolving A::f() in the docs when there was a A::f(int) but also a
1095 // global function f() that exactly matched the argument list.
1096 else if (isParentScope(scope,newScope) && newScope->definitionType()==Definition::TypeClass)
1097 {
1098 // if we a look for a type B and have explicit scope A, then it is also fine if B
1099 // is found at the global scope.
1100 result = 1;
1101 goto done;
1102 }
1103#endif
1104 // repeat for the parent scope
1105 if (scope!=Doxygen::globalScope)
1106 {
1107 i = isAccessibleFromWithExpScope(visitedKeys,visitedNamespaces,accessStack,scope->getOuterScope(),item,explicitScopePart);
1108 }
1109 result = (i==-1) ? -1 : i+2;
1110 }
1111 }
1112 else // failed to resolve explicitScope
1113 {
1114 AUTO_TRACE_ADD("failed to resolve explicitScope");
1115 if (scope->definitionType()==Definition::TypeNamespace)
1116 {
1117 const NamespaceDef *nscope = toNamespaceDef(scope);
1118 VisitedNamespaceKeys locVisitedNamespaceKeys;
1119 if (accessibleViaUsingNamespace(visitedKeys,locVisitedNamespaceKeys,nscope->getUsedNamespaces(),item,explicitScopePart))
1120 {
1121 AUTO_TRACE_ADD("found in used class");
1122 goto done;
1123 }
1124 }
1125 if (scope==Doxygen::globalScope)
1126 {
1127 if (m_fileScope)
1128 {
1129 VisitedNamespaceKeys locVisitedNamespaceKeys;
1130 if (accessibleViaUsingNamespace(visitedKeys,locVisitedNamespaceKeys,m_fileScope->getUsedNamespaces(),item,explicitScopePart))
1131 {
1132 AUTO_TRACE_ADD("found in used namespace");
1133 goto done;
1134 }
1135 }
1136 AUTO_TRACE_ADD("not found in this scope");
1137 result=-1;
1138 }
1139 else // continue by looking into the parent scope
1140 {
1141 int i=isAccessibleFromWithExpScope(visitedKeys,visitedNamespaces,accessStack,scope->getOuterScope(),item,explicitScopePart);
1142 result= (i==-1) ? -1 : i+2;
1143 }
1144 }
1145
1146done:
1147 AUTO_TRACE_EXIT("result={}",result);
1148 accessStack.pop();
1149 return result;
1150}

1152const Definition *SymbolResolver::Private::followPath(VisitedKeys &visitedKeys,
1153 const Definition *start,const QCString &path)
1154{
1155 AUTO_TRACE("start={},path={}",start?start->name():QCString(), path);
1156 int is=0,ps=0,l=0;
1157 const Definition *current=start;
1158 // for each part of the explicit scope
1159 while ((is=getScopeFragment(path,ps,&l))!=-1)
1160 {
1161 // try to resolve the part if it is a typedef
1162 const MemberDef *memTypeDef=nullptr;
1163 QCString qualScopePart = substTypedef(visitedKeys,current,path.mid(is,l),&memTypeDef);
1164 AUTO_TRACE_ADD("qualScopePart={}",qualScopePart);
1165 if (memTypeDef)
1166 {
1167 const ClassDef *type = newResolveTypedef(visitedKeys,m_fileScope,memTypeDef,nullptr,nullptr,nullptr);
1168 if (type)
1169 {
1170 AUTO_TRACE_EXIT("type={}",type->name());
1171 return type;
1172 }
1173 }
1174 const Definition *next = current->findInnerCompound(qualScopePart);
1175 AUTO_TRACE_ADD("Looking for {} inside {} result={}",
1176 qualScopePart, current->name(), next?next->name():QCString());
1177 if (next==nullptr)
1178 {
1179 next = current->findInnerCompound(qualScopePart+"-p");
1180 }
1181 if (current->definitionType()==Definition::TypeClass)
1182 {
1183 const MemberDef *classMember = toClassDef(current)->getMemberByName(qualScopePart);
1184 if (classMember && classMember->isEnumerate())
1185 {
1186 next = classMember;
1187 }
1188 }
1189 else if (current!=Doxygen::globalScope && current->definitionType()==Definition::TypeNamespace)
1190 {
1191 const MemberDef *namespaceMember = toNamespaceDef(current)->getMemberByName(qualScopePart);
1192 if (namespaceMember && namespaceMember->isEnumerate())
1193 {
1194 next = namespaceMember;
1195 }
1196 }
1197 else if (current==Doxygen::globalScope || current->definitionType()==Definition::TypeFile)
1198 {
1199 auto &range = Doxygen::symbolMap->find(qualScopePart);
1200 for (Definition *def : range)
1201 {
1202 const Definition *outerScope = def->getOuterScope();
1203 if (
1204 (outerScope==Doxygen::globalScope || // global scope or
1205 (outerScope && // anonymous namespace in the global scope
1206 outerScope->name().startsWith("anonymous_namespace{") &&
1207 outerScope->getOuterScope()==Doxygen::globalScope
1208 )
1209 ) &&
1210 (def->definitionType()==Definition::TypeClass ||
1211 def->definitionType()==Definition::TypeMember ||
1212 def->definitionType()==Definition::TypeNamespace
1213 )
1214 )
1215 {
1216 next=def;
1217 break;
1218 }
1219 }
1220 }
1221 if (next==nullptr) // failed to follow the path
1222 {
1223 if (current->definitionType()==Definition::TypeNamespace)
1224 {
1225 next = endOfPathIsUsedClass(
1226 (toNamespaceDef(current))->getUsedDefinitions(),qualScopePart);
1227 }
1228 else if (current->definitionType()==Definition::TypeFile)
1229 {
1230 next = endOfPathIsUsedClass(
1231 (toFileDef(current))->getUsedDefinitions(),qualScopePart);
1232 }
1233 current = next;
1234 if (current==nullptr) break;
1235 }
1236 else // continue to follow scope
1237 {
1238 current = next;
1239 AUTO_TRACE_ADD("current={}",current->name());
1240 }
1241 ps=is+l;
1242 }
1243 AUTO_TRACE_EXIT("result={}",current?current->name():QCString());
1244 return current; // path could be followed
1245}

1317int SymbolResolver::Private::isAccessibleFrom(VisitedKeys &visitedKeys,
1318 AccessStack &accessStack,
1319 const Definition *scope,
1320 const Definition *item)
1321{
1322 AUTO_TRACE("scope={} item={}",
1323 scope?scope->name():QCString(), item?item->name():QCString());
1324
1325 if (accessStack.find(scope,m_fileScope,item))
1326 {
1327 AUTO_TRACE_EXIT("already processed!");
1328 return -1;
1329 }
1330 accessStack.push(scope,m_fileScope,item);
1331
1332 int result=0; // assume we found it
1333 int i=0;
1334
1335 const Definition *itemScope=item->getOuterScope();
1336 bool itemIsMember = item->definitionType()==Definition::TypeMember;
1337 bool itemIsClass = item->definitionType()==Definition::TypeClass;
1338
1339 // if item is a global member and scope points to a specific file
1340 // we adjust the scope so the file gets preference over members with the same name in
1341 // other files.
1342 if ((itemIsMember || itemIsClass) &&
1343 (itemScope==Doxygen::globalScope || // global
1344 (itemScope && itemScope->name().startsWith("anonymous_namespace{")) // member of an anonymous namespace
1345 ) &&
1346 scope->definitionType()==Definition::TypeFile)
1347 {
1348 if (itemIsMember)
1349 {
1350 itemScope = toMemberDef(item)->getFileDef();
1351 }
1352 else if (itemIsClass)
1353 {
1354 itemScope = toClassDef(item)->getFileDef();
1355 }
1356 AUTO_TRACE_ADD("adjusting scope to {}",itemScope?itemScope->name():QCString());
1357 }
1358
1359 bool memberAccessibleFromScope =
1360 (itemIsMember && // a member
1361 itemScope && itemScope->definitionType()==Definition::TypeClass && // of a class
1362 scope->definitionType()==Definition::TypeClass && // accessible
1363 (toClassDef(scope))->isAccessibleMember(toMemberDef(item)) // from scope
1364 );
1365 bool nestedClassInsideBaseClass =
1366 (itemIsClass && // a nested class
1367 itemScope && itemScope->definitionType()==Definition::TypeClass && // inside a base
1368 scope->definitionType()==Definition::TypeClass && // class of scope
1369 (toClassDef(scope))->isBaseClass(toClassDef(itemScope),TRUE)
1370 );
1371 bool enumValueOfStrongEnum =
1372 (itemIsMember &&

1375 toMemberDef(scope)->isEnumerate() &&
1376 scope==toMemberDef(item)->getEnumScope()
1377 );

1384 if (nestedClassInsideBaseClass)

1386 result++; // penalty for base class to prevent
1387 // this is preferred over nested class in this class

1389 }

1392 itemScope->definitionType()==Definition::TypeClass &&

1394 (distanceToBase=toClassDef(scope)->isBaseClass(toClassDef(itemScope),TRUE))>0
1395 )

1398 }
1399 }

1428 }

1432 if (scope->definitionType()==Definition::TypeNamespace)

1442 {
1443 AUTO_TRACE_ADD("found via used namespace");

1445 }
1446 }
1447 else if (scope->definitionType()==Definition::TypeFile)
1448 {
1449 const FileDef *nfile = toFileDef(scope);
1450 if (accessibleViaUsingDefinition(visitedKeys,nfile->getUsedDefinitions(),item))
1451 {
1452 AUTO_TRACE_ADD("found via used class");
1453 goto done;
1454 }
1455 VisitedNamespaceKeys visitedNamespaceKeys;
1456 if (accessibleViaUsingNamespace(visitedKeys,visitedNamespaceKeys,nfile->getUsedNamespaces(),item,QCString()))
1457 {
1458 AUTO_TRACE_ADD("found via used namespace");
1459 goto done;
1460 }
1461 }
1462 // repeat for the parent scope
1463 const Definition *parentScope = scope->getOuterScope();

1465 {

1470 {

1472 }
1473 }
1474 }
1475 i=isAccessibleFrom(visitedKeys,accessStack,parentScope,item);
1476 result= (i==-1) ? -1 : i+2;
1477 }
1478done:
1479 AUTO_TRACE_EXIT("result={}",result);
1480 accessStack.pop();
1481 return result;
1482}
