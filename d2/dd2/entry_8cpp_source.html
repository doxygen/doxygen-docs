170void Entry::removeSubEntry(const Entry *e)
171{
172 auto it = std::find_if(m_sublist.begin(),m_sublist.end(),
173 [e](const std::shared_ptr<Entry>&elem) { return elem.get()==e; });
174 if (it!=m_sublist.end())
175 {
176 m_sublist.erase(it);
177 }
178}
