45{
46 for (const auto &attr : cell.attribs())

167void RTFDocVisitor::operator()(const DocEmoji &s)
168{
169 if (m_hide) return;
170 DBG_RTF("{\\comment RTFDocVisitor::visit(DocEmoji)}\n");
171 const char *res = EmojiEntityMapper::instance().unicode(s.index());
172 if (res)
173 {
174 const char *p = res;
175 int val = 0;
176 int val1 = 0;
177 while (*p)
178 {
179 switch(*p)
180 {
181 case '&': case '#': case 'x':
182 break;
183 case ';':
184 val1 = val;
185 val = 0xd800 + ( ( val1 - 0x10000 ) & 0xffc00 ) / 0x400 - 0x10000;
186 m_t << "\\u" << val << "?";
187 val = 0xdC00 + ( ( val1 - 0x10000 ) & 0x3ff ) - 0x10000 ;
188 m_t << "\\u" << val << "?";
189 val = 0;
190 break;
191 case '0': case '1': case '2': case '3': case '4':
192 case '5': case '6': case '7': case '8': case '9':
193 val = val * 16 + *p - '0';
194 break;
195 case 'a': case 'b': case 'c': case 'd': case 'e': case 'f':
196 val = val * 16 + *p - 'a' + 10;
197 break;
198 }
199 p++;
200 }
201 }
202 else
203 {
204 m_t << s.name();
205 }
206 m_lastIsPara=FALSE;
207}

316void RTFDocVisitor::operator()(const DocVerbatim &s)
317{
318 if (m_hide) return;
319 DBG_RTF("{\\comment RTFDocVisitor::visit(DocVerbatim)}\n");
320 QCString lang = m_langExt;
321 if (!s.language().isEmpty()) // explicit language setting
322 {
323 lang = s.language();
324 }
325 SrcLangExt langExt = getLanguageFromCodeLang(lang);
326 switch(s.type())
327 {
328 case DocVerbatim::Code:
329 m_t << "{\n";
330 m_t << "\\par\n";
331 m_t << rtf_Style_Reset << getStyle("CodeExample");
332 getCodeParser(lang).parseCode(m_ci,s.context(),s.text(),langExt,
333 Config_getBool(STRIP_CODE_COMMENTS),
334 s.isExample(),s.exampleFile());
335 //m_t << "\\par\n";
336 m_t << "}\n";
337 break;
338 case DocVerbatim::JavaDocLiteral:
339 filter(s.text(),TRUE);
340 break;
341 case DocVerbatim::JavaDocCode:
342 m_t << "{\n";
343 m_t << "{\\f2 ";
344 filter(s.text(),TRUE);
345 m_t << "}";
346 m_t << "}\n";
347 break;
348 case DocVerbatim::Verbatim:
349 m_t << "{\n";
350 m_t << "\\par\n";
351 m_t << rtf_Style_Reset << getStyle("CodeExample");
352 filter(s.text(),TRUE);
353 //m_t << "\\par\n";
354 m_t << "}\n";
355 break;
356 case DocVerbatim::RtfOnly:
357 m_t << s.text();
358 break;
359 case DocVerbatim::HtmlOnly:
360 case DocVerbatim::LatexOnly:
361 case DocVerbatim::XmlOnly:
362 case DocVerbatim::ManOnly:
363 case DocVerbatim::DocbookOnly:
364 /* nothing */
365 break;
366 case DocVerbatim::Dot:
367 {
368 static int dotindex = 1;
369 QCString fileName(4096, QCString::ExplicitSize);
370
371 fileName.sprintf("%s%d%s",
372 qPrint(Config_getString(RTF_OUTPUT)+"/inline_dotgraph_"),
373 dotindex++,
374 ".dot"
375 );
376 std::ofstream file = Portable::openOutputStream(fileName);
377 if (!file.is_open())
378 {
379 err("Could not open file %s for writing\n",qPrint(fileName));
380 }
381 else
382 {
383 QCString stext = s.text();
384 file.write( stext.data(), stext.length() );
385 file.close();
386 }
387
388 writeDotFile(fileName, s.hasCaption(), s.srcFile(), s.srcLine());
389 visitChildren(s);
390 includePicturePostRTF(true, s.hasCaption());
391
392 if (Config_getBool(DOT_CLEANUP)) Dir().remove(fileName.str());
393 }
394 break;
395 case DocVerbatim::Msc:
396 {
397 static int mscindex = 1;
398 QCString baseName(4096, QCString::ExplicitSize);
399
400 baseName.sprintf("%s%d%s",
401 qPrint(Config_getString(RTF_OUTPUT)+"/inline_mscgraph_"),
402 mscindex++,
403 ".msc"
404 );
405 std::ofstream file = Portable::openOutputStream(baseName);
406 if (!file.is_open())
407 {
408 err("Could not open file %s for writing\n",qPrint(baseName));
409 }
410 QCString text = "msc {";
411 text+=s.text();
412 text+="}";
413 file.write( text.data(), text.length() );
414 file.close();
415
416 writeMscFile(baseName, s.hasCaption(), s.srcFile(), s.srcLine());
417 visitChildren(s);
418 includePicturePostRTF(true, s.hasCaption());
419
420 if (Config_getBool(DOT_CLEANUP)) Dir().remove(baseName.str());
421 }
422 break;
423 case DocVerbatim::PlantUML:
424 {
425 QCString rtfOutput = Config_getString(RTF_OUTPUT);
426 QCString baseName = PlantumlManager::instance().writePlantUMLSource(
427 rtfOutput,s.exampleFile(),s.text(),PlantumlManager::PUML_BITMAP,
428 s.engine(),s.srcFile(),s.srcLine(),true);
429
430 writePlantUMLFile(baseName, s.hasCaption());
431 visitChildren(s);
432 includePicturePostRTF(true, s.hasCaption());
433 }
434 break;
435 }
436 m_lastIsPara=FALSE;
437}

554 QCString locLangExt = getFileNameExtension(op.includeFileName());

567 }
568 if (op.type()!=DocIncOperator::Skip)
569 {
570 m_hide = popHidden();
571 if (!m_hide)
572 {

574 if (!op.includeFileName().isEmpty())

577 fd = createFileDef( cfi.dirPath(), cfi.fileName() );
578 }
579
580 getCodeParser(locLangExt).parseCode(m_ci,op.context(),op.text(),langExt,
581 op.stripCodeComments(),
582 op.isExample(),op.exampleFile(),
583 fd.get(), // fileDef
584 op.line(), // startLine
585 -1, // endLine
586 FALSE, // inline fragment
587 nullptr, // memberDef
588 op.showLineNo() // show line numbers
589 );
590 }
591 pushHidden(m_hide);
592 m_hide=TRUE;
593 }
594 if (op.isLast())
595 {
596 m_hide = popHidden();
597 if (!m_hide)
598 {
599 m_t << "\\par";
600 m_t << "}\n";
601 }
602 m_lastIsPara=TRUE;
603 }
604 else
605 {
606 if (!m_hide) m_t << "\n";
607 m_lastIsPara=FALSE;
608 }

692 static int prevLevel = -1;
693 if (m_hide) return;
694 DBG_RTF("{\\comment RTFDocVisitor::operator()(const DocAutoListItem &)}\n");
695 int level = indentLevel();
696 if ((level != prevLevel-1) &&
697 (!(level==prevLevel && level != 0 && m_listItemInfo[level].isCheck)) &&
698 (!m_lastIsPara))
699 m_t << "\\par\n";
700 prevLevel= level;
701 m_t << rtf_Style_Reset;
702 if (m_listItemInfo[level].isEnum)
703 {
704 m_t << getStyle("ListEnum") << "\n";
705 m_t << m_listItemInfo[level].number << ".\\tab ";
706 m_listItemInfo[level].number++;
707 }
708 else
709 {
710 switch (li.itemNumber())
711 {
712 case DocAutoList::Unchecked: // unchecked
713 m_t << getListTable(2) << "\n";
714 break;
715 case DocAutoList::Checked_x: // checked with x
716 case DocAutoList::Checked_X: // checked with X
717 m_t << getListTable(3) << "\n";
718 break;
719 default:
720 m_t << getListTable(1) << "\n";
721 break;
722 }
723 }
724 incIndentLevel();
725 m_lastIsPara=false;
726 visitChildren(li);
727 decIndentLevel();
728}
