204// test if the next characters in data represent a new line (which can be character \n or string \ilinebr).
205// returns 0 if no newline is found, or the number of characters that make up the newline if found.
206inline size_t isNewline(std::string_view data)
207{
208 // normal newline
209 if (data[0] == '\n') return 1;
210 // artificial new line from ^^ in ALIASES
211 if (data[0] == '\\' && qstrncmp(data.data()+1,"ilinebr ",7)==0) return data[8]==' ' ? 9 : 8;
212 return 0;
213}

215// escape double quotes in string
216static QCString escapeDoubleQuotes(const QCString &s)
217{
218 AUTO_TRACE("s={}",Trace::trunc(s));
219 if (s.isEmpty()) return s;
220 QCString result;
221 const char *p=s.data();
222 char c=0, pc='\0';
223 while ((c=*p++))
224 {
225 if (c=='"' && pc!='\\') result+='\\';
226 result+=c;
227 pc=c;
228 }
229 AUTO_TRACE_EXIT("result={}",result);
230 return result;
231}

233// escape characters that have a special meaning later on.
234static QCString escapeSpecialChars(const QCString &s)
235{
236 AUTO_TRACE("s={}",Trace::trunc(s));
237 if (s.isEmpty()) return s;
238 bool insideQuote=FALSE;
239 QCString result;
240 const char *p=s.data();
241 char c=0, pc='\0';
242 while ((c=*p++))
243 {
244 switch (c)
245 {
246 case '"':
247 if (pc!='\\') { insideQuote=!insideQuote; }
248 result+=c;
249 break;
250 case '<':
251 // fall through
252 case '>':
253 if (!insideQuote)
254 {
255 result+='\\';
256 result+=c;
257 if ((p[0]==':') && (p[1]==':'))
258 {
259 result+='\\';
260 result+=':';
261 p++;
262 }
263 }
264 else
265 {
266 result+=c;
267 }
268 break;
269 case '\\': if (!insideQuote) { result+='\\'; } result+='\\'; break;
270 case '@': if (!insideQuote) { result+='\\'; } result+='@'; break;
271 // commented out next line due to regression when using % to suppress a link
272 //case '%': if (!insideQuote) { result+='\\'; } result+='%'; break;
273 case '#': if (!insideQuote) { result+='\\'; } result+='#'; break;
274 case '$': if (!insideQuote) { result+='\\'; } result+='$'; break;
275 case '&': if (!insideQuote) { result+='\\'; } result+='&'; break;
276 default:
277 result+=c; break;
278 }
279 pc=c;
280 }
281 AUTO_TRACE_EXIT("result={}",result);
282 return result;
283}

335// Check if data contains a block command. If so returned the command
336// that ends the block. If not an empty string is returned.
337// Note When offset>0 character position -1 will be inspected.
338//
339// Checks for and skip the following block commands:
340// {@code .. { .. } .. }
341// \dot .. \enddot
342// \code .. \endcode
343// \msc .. \endmsc
344// \f$..\f$
345// \f(..\f)
346// \f[..\f]
347// \f{..\f}
348// \verbatim..\endverbatim
349// \iliteral..\endiliteral
350// \latexonly..\endlatexonly
351// \htmlonly..\endhtmlonly
352// \xmlonly..\endxmlonly
353// \rtfonly..\endrtfonly
354// \manonly..\endmanonly
355// \startuml..\enduml
356QCString Markdown::Private::isBlockCommand(std::string_view data,size_t offset)
357{
358 AUTO_TRACE("data='{}' offset={}",Trace::trunc(data),offset);
359
360 using EndBlockFunc = QCString (*)(const std::string &,bool,char);
361
362 static const auto getEndBlock = [](const std::string &blockName,bool,char) -> QCString
363 {
364 return "end"+blockName;
365 };
366 static const auto getEndCode = [](const std::string &blockName,bool openBracket,char) -> QCString
367 {
368 return openBracket ? QCString("}") : "end"+blockName;
369 };
370 static const auto getEndUml = [](const std::string &/* blockName */,bool,char) -> QCString
371 {
372 return "enduml";
373 };
374 static const auto getEndFormula = [](const std::string &/* blockName */,bool,char nextChar) -> QCString
375 {
376 switch (nextChar)
377 {
378 case '$': return "f$";
379 case '(': return "f)";
380 case '[': return "f]";
381 case '{': return "f}";
382 }
383 return "";
384 };
385
386 // table mapping a block start command to a function that can return the matching end block string
387 static const std::unordered_map<std::string,EndBlockFunc> blockNames =
388 {
389 { "dot", getEndBlock },
390 { "code", getEndCode },
391 { "icode", getEndBlock },
392 { "msc", getEndBlock },
393 { "verbatim", getEndBlock },
394 { "iverbatim", getEndBlock },
395 { "iliteral", getEndBlock },
396 { "latexonly", getEndBlock },
397 { "htmlonly", getEndBlock },
398 { "xmlonly", getEndBlock },
399 { "rtfonly", getEndBlock },
400 { "manonly", getEndBlock },
401 { "docbookonly", getEndBlock },
402 { "startuml", getEndUml },
403 { "f", getEndFormula }
404 };
405
406 const size_t size = data.size();
407 bool openBracket = offset>0 && data.data()[-1]=='{';
408 bool isEscaped = offset>0 && (data.data()[-1]=='\\' || data.data()[-1]=='@');
409 if (isEscaped) return QCString();
410
411 size_t end=1;
412 while (end<size && (data[end]>='a' && data[end]<='z')) end++;
413 if (end==1) return QCString();
414 std::string blockName(data.substr(1,end-1));
415 auto it = blockNames.find(blockName);
416 QCString result;
417 if (it!=blockNames.end()) // there is a function assigned
418 {
419 result = it->second(blockName, openBracket, end<size ? data[end] : 0);
420 }
421 AUTO_TRACE_EXIT("result={}",result);
422 return result;
423}

431 static const auto endOfLine = [](std::string_view data_,size_t offset_) -> size_t
432 {
433 // skip until the end of line (allowing line continuation characters)
434 char lc = 0;
435 char c = 0;
436 while (offset_<data_.size() && ((c=data_[offset_])!='\n' || lc=='\\'))
437 {
438 if (c=='\\') lc='\\'; // last character was a line continuation
439 else if (c!=' ') lc=0; // rest line continuation
440 offset_++;
441 }
442 return offset_;
443 };
444
445 static const auto endOfLabel = [](std::string_view data_,size_t offset_) -> size_t
446 {
447 if (offset_<data_.size() && data_[offset_]==' ') // we expect a space before the label
448 {
449 char c = 0;
450 offset_++;
451 // skip over spaces
452 while (offset_<data_.size() && data_[offset_]==' ') offset_++;
453 // skip over label
454 while (offset_<data_.size() && (c=data_[offset_])!=' ' && c!='\\' && c!='@' && c!='\n') offset_++;
455 return offset_;
456 }
457 return 0;
458 };

460 static const auto endOfLabelOpt = [](std::string_view data_,size_t offset_) -> size_t
461 {
462 size_t index=offset_;
463 if (index<data_.size() && data_[index]==' ') // skip over optional spaces
464 {
465 index++;
466 while (index<data_.size() && data_[index]==' ') index++;
467 }
468 if (index<data_.size() && data_[index]=='{') // find matching '}'
469 {
470 index++;
471 char c = 0;
472 while (index<data_.size() && (c=data_[index])!='}' && c!='\\' && c!='@' && c!='\n') index++;
473 if (index==data_.size() || data_[index]!='}') return 0; // invalid option
474 offset_=index+1; // part after {...} is the option
475 }
476 return endOfLabel(data_,offset_);
477 };
478
479 static const auto endOfParam = [](std::string_view data_,size_t offset_) -> size_t
480 {
481 size_t index=offset_;
482 if (index<data_.size() && data_[index]==' ') // skip over optional spaces
483 {
484 index++;
485 while (index<data_.size() && data_[index]==' ') index++;
486 }
487 if (index<data_.size() && data_[index]=='[') // find matching ']'
488 {
489 index++;
490 char c = 0;
491 while (index<data_.size() && (c=data_[index])!=']' && c!='\n') index++;
492 if (index==data_.size() || data_[index]!=']') return 0; // invalid parameter
493 offset_=index+1; // part after [...] is the parameter name
494 }
495 return endOfLabel(data_,offset_);
496 };
497
498 static const auto endOfFuncLike = [](std::string_view data_,size_t offset_,bool allowSpaces) -> size_t
499 {
500 if (offset_<data_.size() && data_[offset_]==' ') // we expect a space before the name
501 {
502 char c=0;
503 offset_++;
504 // skip over spaces
505 while (offset_<data_.size() && data_[offset_]==' ')
506 {
507 offset_++;
508 }
509 // skip over name (and optionally type)
510 while (offset_<data_.size() && (c=data_[offset_])!='\n' && (allowSpaces || c!=' ') && c!='(')
511 {
512 offset_++;
513 }
514 if (c=='(') // find the end of the function
515 {
516 int count=1;
517 offset_++;
518 while (offset_<data_.size() && (c=data_[offset_++]))
519 {
520 if (c=='(') count++;
521 else if (c==')') count--;
522 if (count==0) return offset_;
523 }
524 }
525 return offset_;
526 }
527 return 0;
528 };

653/** looks for the next emph char, skipping other constructs, and
654 * stopping when either it is found, or we are at the end of a paragraph.
655 */
656size_t Markdown::Private::findEmphasisChar(std::string_view data, char c, size_t c_size)
657{
658 AUTO_TRACE("data='{}' c={} c_size={}",Trace::trunc(data),c,c_size);
659 size_t i = 1;
660 const size_t size = data.size();
661
662 while (i<size)
663 {
664 while (i<size && data[i]!=c &&
665 data[i]!='\\' && data[i]!='@' &&
666 !(data[i]=='/' && data[i-1]=='<') && // html end tag also ends emphasis
667 data[i]!='\n') i++;
668 // avoid overflow (unclosed emph token)
669 if (i==size)
670 {
671 return 0;
672 }
673 //printf("findEmphasisChar: data=[%s] i=%d c=%c\n",data,i,data[i]);
674
675 // not counting escaped chars or characters that are unlikely
676 // to appear as the end of the emphasis char
677 if (ignoreCloseEmphChar(data[i-1],data[i]))
678 {
679 i++;
680 continue;
681 }
682 else
683 {
684 // get length of emphasis token
685 size_t len = 0;
686 while (i+len<size && data[i+len]==c)
687 {
688 len++;
689 }
690
691 if (len>0)
692 {
693 if (len!=c_size || (i+len<size && isIdChar(data[i+len]))) // to prevent touching some_underscore_identifier
694 {
695 i+=len;
696 continue;
697 }
698 AUTO_TRACE_EXIT("result={}",i);
699 return static_cast<int>(i); // found it
700 }
701 }
702
703 // skipping a code span
704 if (data[i]=='`')
705 {
706 int snb=0;
707 while (i<size && data[i]=='`') snb++,i++;
708
709 // find same pattern to end the span
710 int enb=0;
711 while (i<size && enb<snb)
712 {
713 if (data[i]=='`') enb++;
714 if (snb==1 && data[i]=='\'') break; // ` ended by '
715 i++;
716 }
717 }
718 else if (data[i]=='@' || data[i]=='\\')
719 { // skip over blocks that should not be processed
720 QCString endBlockName = isBlockCommand(data.substr(i),i);
721 if (!endBlockName.isEmpty())
722 {
723 i++;
724 size_t l = endBlockName.length();
725 while (i+l<size)
726 {
727 if ((data[i]=='\\' || data[i]=='@') && // command
728 data[i-1]!='\\' && data[i-1]!='@') // not escaped
729 {
730 if (qstrncmp(&data[i+1],endBlockName.data(),l)==0)
731 {
732 break;
733 }
734 }
735 i++;
736 }
737 }
738 else if (i+1<size && isIdChar(data[i+1])) // @cmd, stop processing, see bug 690385
739 {
740 return 0;
741 }
742 else
743 {
744 i++;
745 }
746 }
747 else if (data[i-1]=='<' && data[i]=='/') // html end tag invalidates emphasis
748 {
749 return 0;
750 }
751 else if (data[i]=='\n') // end * or _ at paragraph boundary
752 {
753 i++;
754 while (i<size && data[i]==' ') i++;
755 if (i>=size || data[i]=='\n')
756 {
757 return 0;
758 } // empty line -> paragraph
759 }
760 else // should not get here!
761 {
762 i++;
763 }
764 }
765 return 0;
766}

896/** Process ndash and mdashes */
897int Markdown::Private::processNmdash(std::string_view data,size_t offset)
898{
899 AUTO_TRACE("data='{}' offset={}",Trace::trunc(data),offset);
900 const size_t size = data.size();

902 size_t i=1;
903 int count=1;
904 if (i<size && data[i]=='-') // found --
905 {
906 count++,i++;
907 }
908 if (i<size && data[i]=='-') // found ---
909 {
910 count++,i++;
911 }
912 if (i<size && data[i]=='-') // found ----
913 {
914 count++;
915 }
916 if (count>=2 && offset>=2 && qstrncmp(data.data()-2,"<!",2)==0)
917 { AUTO_TRACE_EXIT("result={}",1-count); return 1-count; } // start HTML comment
918 if (count==2 && size > 2 && data[2]=='>')
919 { return 0; } // end HTML comment
920 if (count==3 && size > 3 && data[3]=='>')
921 { return 0; } // end HTML comment
922 if (count==2 && (offset<8 || qstrncmp(data.data()-8,"operator",8)!=0)) // -- => ndash
923 {
924 out+="&ndash;";
925 AUTO_TRACE_EXIT("result=2");
926 return 2;
927 }
928 else if (count==3) // --- => ndash
929 {
930 out+="&mdash;";
931 AUTO_TRACE_EXIT("result=3");
932 return 3;
933 }
934 // not an ndash or mdash
935 return 0;

960/** Process a HTML tag. Note that <pre>..</pre> are treated specially, in
961 * the sense that all code inside is written unprocessed
962 */
963int Markdown::Private::processHtmlTagWrite(std::string_view data,size_t offset,bool doWrite)
964{
965 AUTO_TRACE("data='{}' offset={} doWrite={}",Trace::trunc(data),offset,doWrite);
966 if (offset>0 && data.data()[-1]=='\\') { return 0; } // escaped <
967
968 const size_t size = data.size();
969
970 // find the end of the html tag
971 size_t i=1;
972 size_t l=0;
973 // compute length of the tag name
974 while (i<size && isIdChar(data[i])) i++,l++;
975 QCString tagName(data.substr(1,i-1));
976 if (tagName.lower()=="pre") // found <pre> tag
977 {
978 bool insideStr=FALSE;
979 while (i+6<size)
980 {
981 char c=data[i];
982 if (!insideStr && c=='<') // potential start of html tag
983 {
984 if (data[i+1]=='/' &&
985 tolower(data[i+2])=='p' && tolower(data[i+3])=='r' &&
986 tolower(data[i+4])=='e' && tolower(data[i+5])=='>')
987 { // found </pre> tag, copy from start to end of tag
988 if (doWrite) out+=data.substr(0,i+6);
989 //printf("found <pre>..</pre> [%d..%d]\n",0,i+6);
990 AUTO_TRACE_EXIT("result={}",i+6);
991 return static_cast<int>(i+6);
992 }
993 }
994 else if (insideStr && c=='"')
995 {
996 if (data[i-1]!='\\') insideStr=FALSE;
997 }
998 else if (c=='"')
999 {
1000 insideStr=TRUE;
1001 }
1002 i++;
1003 }
1004 }
1005 else // some other html tag
1006 {
1007 if (l>0 && i<size)
1008 {
1009 if (data[i]=='/' && i<size-1 && data[i+1]=='>') // <bla/>
1010 {
1011 //printf("Found htmlTag={%s}\n",qPrint(QCString(data).left(i+2)));
1012 if (doWrite) out+=data.substr(0,i+2);
1013 AUTO_TRACE_EXIT("result={}",i+2);
1014 return static_cast<int>(i+2);
1015 }
1016 else if (data[i]=='>') // <bla>
1017 {
1018 //printf("Found htmlTag={%s}\n",qPrint(QCString(data).left(i+1)));
1019 if (doWrite) out+=data.substr(0,i+1);
1020 AUTO_TRACE_EXIT("result={}",i+1);
1021 return static_cast<int>(i+1);
1022 }
1023 else if (data[i]==' ') // <bla attr=...
1024 {
1025 i++;
1026 bool insideAttr=FALSE;
1027 while (i<size)
1028 {
1029 if (!insideAttr && data[i]=='"')
1030 {
1031 insideAttr=TRUE;
1032 }
1033 else if (data[i]=='"' && data[i-1]!='\\')
1034 {
1035 insideAttr=FALSE;
1036 }
1037 else if (!insideAttr && data[i]=='>') // found end of tag
1038 {
1039 //printf("Found htmlTag={%s}\n",qPrint(QCString(data).left(i+1)));
1040 if (doWrite) out+=data.substr(0,i+1);
1041 AUTO_TRACE_EXIT("result={}",i+1);
1042 return static_cast<int>(i+1);
1043 }
1044 i++;
1045 }
1046 }
1047 }
1048 }
1049 AUTO_TRACE_EXIT("not a valid html tag");
1050 return 0;
1051}

1059int Markdown::Private::processEmphasis(std::string_view data,size_t offset)
1060{
1061 AUTO_TRACE("data='{}' offset={}",Trace::trunc(data),offset);
1062 const size_t size = data.size();
1063
1064 if ((offset>0 && !isOpenEmphChar(data.data()[-1])) || // invalid char before * or _
1065 (size>1 && data[0]!=data[1] && !(isIdChar(data[1]) || extraChar(data[1]))) || // invalid char after * or _
1066 (size>2 && data[0]==data[1] && !(isIdChar(data[2]) || extraChar(data[2])))) // invalid char after ** or __
1067 {
1068 AUTO_TRACE_EXIT("invalid surrounding characters");
1069 return 0;
1070 }
1071
1072 char c = data[0];
1073 int ret = 0;
1074 if (size>2 && c!='~' && data[1]!=c) // _bla or *bla
1075 {
1076 // whitespace cannot follow an opening emphasis
1077 if (data[1]==' ' || data[1]=='\n' ||
1078 (ret = processEmphasis1(data.substr(1), c)) == 0)
1079 {
1080 return 0;
1081 }
1082 AUTO_TRACE_EXIT("result={}",ret+1);
1083 return ret+1;
1084 }
1085 if (size>3 && data[1]==c && data[2]!=c) // __bla or **bla
1086 {
1087 if (data[2]==' ' || data[2]=='\n' ||
1088 (ret = processEmphasis2(data.substr(2), c)) == 0)
1089 {
1090 return 0;
1091 }
1092 AUTO_TRACE_EXIT("result={}",ret+2);
1093 return ret+2;
1094 }
1095 if (size>4 && c!='~' && data[1]==c && data[2]==c && data[3]!=c) // ___bla or ***bla
1096 {
1097 if (data[3]==' ' || data[3]=='\n' ||
1098 (ret = processEmphasis3(data.substr(3), c)) == 0)
1099 {
1100 return 0;
1101 }
1102 AUTO_TRACE_EXIT("result={}",ret+3);
1103 return ret+3;
1104 }
1105 return 0;

1582/** `` ` `` parsing a code span (assuming codespan != 0) */
1583int Markdown::Private::processCodeSpan(std::string_view data,size_t)
1584{
1585 AUTO_TRACE("data='{}'",Trace::trunc(data));
1586 const size_t size = data.size();
1587
1588 /* counting the number of backticks in the delimiter */
1589 size_t nb=0, end=0;
1590 while (nb<size && data[nb]=='`')
1591 {
1592 nb++;
1593 }
1594
1595 /* finding the next delimiter */
1596 size_t i = 0;
1597 char pc = '`';
1598 for (end=nb; end<size && i<nb; end++)
1599 {
1600 if (data[end]=='`')
1601 {
1602 i++;
1603 }
1604 else if (data[end]=='\n')
1605 {
1606 // consecutive newlines
1607 if (pc == '\n') return 0;
1608 pc = '\n';
1609 i = 0;
1610 }
1611 else if (data[end]=='\'' && nb==1 && (end==size-1 || (end<size-1 && !isIdChar(data[end+1]))))
1612 { // look for quoted strings like 'some word', but skip strings like `it's cool`
1613 out+="&lsquo;";
1614 out+=data.substr(nb,end-nb);
1615 out+="&rsquo;";
1616 return static_cast<int>(end+1);
1617 }
1618 else
1619 {
1620 if (data[end]!=' ') pc = data[end];
1621 i=0;
1622 }
1623 }
1624 if (i < nb && end >= size)
1625 {
1626 return 0; // no matching delimiter
1627 }
1628
1629 // trimming outside whitespaces
1630 size_t f_begin = nb;
1631 while (f_begin < end && data[f_begin]==' ')
1632 {
1633 f_begin++;
1634 }
1635 size_t f_end = end - nb;
1636 while (f_end > nb && data[f_end-1]==' ')
1637 {
1638 f_end--;
1639 }
1640
1641 //printf("found code span '%s'\n",qPrint(QCString(data+f_begin).left(f_end-f_begin)));
1642
1643 /* real code span */
1644 if (f_begin < f_end)
1645 {
1646 QCString codeFragment = data.substr(f_begin, f_end-f_begin);
1647 out+="<tt>";
1648 out+=escapeSpecialChars(codeFragment);
1649 out+="</tt>";
1650 }
1651 AUTO_TRACE_EXIT("result={}",end);
1652 return static_cast<int>(end);
1653}

1670 AUTO_TRACE("{}",Trace::trunc(data));
1671 const size_t size = data.size();
1672 size_t i=1;
1673 QCString endBlockName = isBlockCommand(data,offset);
1674 if (!endBlockName.isEmpty())
1675 {
1676 AUTO_TRACE_ADD("endBlockName={}",endBlockName);
1677 size_t l = endBlockName.length();

1679 {

1691 i++;
1692 }
1693 }

1695 if (endPos>0)
1696 {
1697 out+=data.substr(0,endPos);

1708 }

1718 AUTO_TRACE_EXIT("2");

1722 {
1723 out+=data.substr(1,2);
1724 AUTO_TRACE_EXIT("3");

1726 }

1733 out+=data.substr(0,2);

1736 }

1742{
1743 AUTO_TRACE("data='{}'",Trace::trunc(data));
1744 size_t i=0;
1745 size_t end=0;
1746 Action_t action;
1747 const size_t size = data.size();
1748 while (i<size)
1749 {
1750 // skip over characters that do not trigger a specific action
1751 while (end<size && ((action=actions[static_cast<uint8_t>(data[end])])==nullptr)) end++;
1752 // and add them to the output
1753 out+=data.substr(i,end-i);

1757 int iend = action(data.substr(i),i);

1759 {
1760 end=i+1-iend;
1761 }
1762 else // skip until end
1763 {
1764 i+=iend;
1765 end=i;
1766 }

1770/** returns whether the line is a setext-style hdr underline */
1771int Markdown::Private::isHeaderline(std::string_view data, bool allowAdjustLevel)
1772{
1773 AUTO_TRACE("data='{}' allowAdjustLevel",Trace::trunc(data),allowAdjustLevel);
1774 size_t i=0, c=0;
1775 const size_t size = data.size();
1776 while (i<size && data[i]==' ') i++;
1777 if (i==size) return 0;
1778
1779 // test of level 1 header
1780 if (data[i]=='=')
1781 {
1782 while (i<size && data[i]=='=') i++,c++;
1783 while (i<size && data[i]==' ') i++;
1784 int level = (c>1 && (i>=size || data[i]=='\n')) ? 1 : 0;
1785 if (allowAdjustLevel && level==1 && indentLevel==-1)
1786 {
1787 // In case a page starts with a header line we use it as title, promoting it to @page.
1788 // We set g_indentLevel to -1 to promoting the other sections if they have a deeper
1789 // nesting level than the page header, i.e. @section..@subsection becomes @page..@section.
1790 // In case a section at the same level is found (@section..@section) however we need
1791 // to undo this (and the result will be @page..@section).
1792 indentLevel=0;
1793 }
1794 AUTO_TRACE_EXIT("result={}",indentLevel+level);
1795 return indentLevel+level;
1796 }
1797 // test of level 2 header
1798 if (data[i]=='-')
1799 {
1800 while (i<size && data[i]=='-') i++,c++;
1801 while (i<size && data[i]==' ') i++;
1802 return (c>1 && (i>=size || data[i]=='\n')) ? indentLevel+2 : 0;
1803 }
1804 return 0;
1805}

1836/** returns end of the link ref if this is indeed a link reference. */
1837static size_t isLinkRef(std::string_view data, QCString &refid, QCString &link, QCString &title)
1838{
1839 AUTO_TRACE("data='{}'",Trace::trunc(data));
1840 const size_t size = data.size();
1841 // format: start with [some text]:
1842 size_t i = 0;
1843 while (i<size && data[i]==' ') i++;
1844 if (i>=size || data[i]!='[') { return 0; }
1845 i++;
1846 size_t refIdStart=i;
1847 while (i<size && data[i]!='\n' && data[i]!=']') i++;
1848 if (i>=size || data[i]!=']') { return 0; }
1849 refid = data.substr(refIdStart,i-refIdStart);
1850 if (refid.isEmpty()) { return 0; }
1851 AUTO_TRACE_ADD("refid found {}",refid);
1852 //printf(" isLinkRef: found refid='%s'\n",qPrint(refid));
1853 i++;
1854 if (i>=size || data[i]!=':') { return 0; }
1855 i++;
1856
1857 // format: whitespace* \n? whitespace* (<url> | url)
1858 while (i<size && data[i]==' ') i++;
1859 if (i<size && data[i]=='\n')
1860 {
1861 i++;
1862 while (i<size && data[i]==' ') i++;
1863 }
1864 if (i>=size) { return 0; }
1865
1866 if (i<size && data[i]=='<') i++;
1867 size_t linkStart=i;
1868 while (i<size && data[i]!=' ' && data[i]!='\n') i++;
1869 size_t linkEnd=i;
1870 if (i<size && data[i]=='>') i++;
1871 if (linkStart==linkEnd) { return 0; } // empty link
1872 link = data.substr(linkStart,linkEnd-linkStart);
1873 AUTO_TRACE_ADD("link found {}",Trace::trunc(link));
1874 if (link=="@ref" || link=="\\ref")
1875 {
1876 size_t argStart=i;
1877 while (i<size && data[i]!='\n' && data[i]!='"') i++;
1878 link+=data.substr(argStart,i-argStart);
1879 }
1880
1881 title.clear();
1882
1883 // format: (whitespace* \n? whitespace* ( 'title' | "title" | (title) ))?
1884 size_t eol=0;
1885 while (i<size && data[i]==' ') i++;
1886 if (i<size && data[i]=='\n')
1887 {
1888 eol=i;
1889 i++;
1890 while (i<size && data[i]==' ') i++;
1891 }
1892 if (i>=size)
1893 {
1894 AUTO_TRACE_EXIT("result={}: end of isLinkRef while looking for title",i);
1895 return i; // end of buffer while looking for the optional title
1896 }
1897
1898 char c = data[i];
1899 if (c=='\'' || c=='"' || c=='(') // optional title present?
1900 {
1901 //printf(" start of title found! char='%c'\n",c);
1902 i++;
1903 if (c=='(') c=')'; // replace c by end character
1904 size_t titleStart=i;
1905 // search for end of the line
1906 while (i<size && data[i]!='\n') i++;
1907 eol = i;
1908
1909 // search back to matching character
1910 size_t end=i-1;
1911 while (end>titleStart && data[end]!=c) end--;
1912 if (end>titleStart)
1913 {
1914 title = data.substr(titleStart,end-titleStart);
1915 }
1916 AUTO_TRACE_ADD("title found {}",Trace::trunc(title));
1917 }
1918 while (i<size && data[i]==' ') i++;
1919 //printf("end of isLinkRef: i=%d size=%d data[i]='%c' eol=%d\n",
1920 // i,size,data[i],eol);
1921 if (i>=size) { AUTO_TRACE_EXIT("result={}",i); return i; } // end of buffer while ref id was found
1922 else if (eol>0) { AUTO_TRACE_EXIT("result={}",eol); return eol; } // end of line while ref id was found
1923 return 0; // invalid link ref
1924}

1990int Markdown::Private::isAtxHeader(std::string_view data,
1991 QCString &header,QCString &id,bool allowAdjustLevel,bool *pIsIdGenerated)
1992{
1993 AUTO_TRACE("data='{}' header={} id={} allowAdjustLevel={}",Trace::trunc(data),Trace::trunc(header),id,allowAdjustLevel);
1994 size_t i = 0;
1995 int level = 0, blanks=0;
1996 const size_t size = data.size();
1997
1998 // find start of header text and determine heading level
1999 while (i<size && data[i]==' ') i++;
2000 if (i>=size || data[i]!='#')
2001 {
2002 return 0;
2003 }
2004 while (i<size && data[i]=='#') i++,level++;
2005 if (level>SectionType::MaxLevel) // too many #'s -> no section
2006 {
2007 return 0;
2008 }
2009 while (i<size && data[i]==' ') i++,blanks++;
2010 if (level==1 && blanks==0)
2011 {
2012 return 0; // special case to prevent #someid seen as a header (see bug 671395)
2013 }
2014
2015 // find end of header text
2016 size_t end=i;
2017 while (end<size && data[end]!='\n') end++;
2018 while (end>i && (data[end-1]=='#' || data[end-1]==' ')) end--;
2019
2020 // store result
2021 header = data.substr(i,end-i);
2022 id = extractTitleId(header, level, pIsIdGenerated);
2023 if (!id.isEmpty()) // strip #'s between title and id
2024 {
2025 int idx=static_cast<int>(header.length())-1;
2026 while (idx>=0 && (header.at(idx)=='#' || header.at(idx)==' ')) idx--;
2027 header=header.left(idx+1);
2028 }
2029
2030 if (allowAdjustLevel && level==1 && indentLevel==-1)
2031 {
2032 // in case we find a `# Section` on a markdown page that started with the same level
2033 // header, we no longer need to artificially decrease the paragraph level.
2034 // So both
2035 // -------------------
2036 // # heading 1 <-- here we set g_indentLevel to -1
2037 // # heading 2 <-- here we set g_indentLevel back to 0 such that this will be a @section
2038 // -------------------
2039 // and
2040 // -------------------
2041 // # heading 1 <-- here we set g_indentLevel to -1
2042 // ## heading 2 <-- here we keep g_indentLevel at -1 such that @subsection will be @section
2043 // -------------------
2044 // will convert to
2045 // -------------------
2046 // @page md_page Heading 1
2047 // @section autotoc_md1 Heading 2
2048 // -------------------
2049
2050 indentLevel=0;
2051 }
2052 int res = level+indentLevel;
