190void PerlModOutput::incIndent()
191{
192 if (m_indentation < PERLOUTPUT_MAX_INDENTATION)
193 {
194 char *s = &m_spaces[m_indentation * 2];
195 *s++ = ' '; *s++ = ' '; *s = 0;

207void PerlModOutput::iaddQuoted(const QCString &str)
208{
209 if (str.isEmpty()) return;
210 const char *s = str.data();
211 char c = 0;
212 while ((c = *s++) != 0)
213 {
214 if ((c == '\'') || (c == '\\'))
215 {
216 m_stream->add('\\');
217 }
218 m_stream->add(c);
219 }
220}

223{
224 continueBlock();
225 m_stream->add(s);
226 m_stream->add(m_pretty ? " => " : "=>");
227}

230{
231 iaddField(field);
232 m_stream->add('\'');
233 if ((content == '\'') || (content == '\\'))
234 m_stream->add('\\');
235 m_stream->add(content);
236 m_stream->add('\'');
237}

239void PerlModOutput::iaddFieldQuotedString(const QCString &field, const QCString &content)
240{
241 if (content == nullptr)
242 return;
243 iaddField(field);
244 m_stream->add('\'');
245 iaddQuoted(content);
246 m_stream->add('\'');

249void PerlModOutput::iopen(char c, const QCString &s)
250{
251 if (s != nullptr)
252 iaddField(s);
253 else
254 continueBlock();
255 m_stream->add(c);
256 incIndent();
257 m_blockstart = true;
258}

260void PerlModOutput::iclose(char c)
261{
262 decIndent();
263 indent();
264 if (c != 0)
265 m_stream->add(c);
266 m_blockstart = false;

494void PerlModDocVisitor::operator()(const DocSymbol &sy)
495{
496 const HtmlEntityMapper::PerlSymb *res = HtmlEntityMapper::instance().perl(sy.symbol());
497 const char *accent=nullptr;
498 if (res->symb)
499 {
500 switch (res->type)
501 {
502 case HtmlEntityMapper::Perl_string:
503 enterText();
504 m_output.add(res->symb);
505 break;
506 case HtmlEntityMapper::Perl_char:
507 enterText();
508 m_output.add(res->symb[0]);
509 break;
510 case HtmlEntityMapper::Perl_symbol:
511 leaveText();
512 openItem("symbol");
513 m_output.addFieldQuotedString("symbol", res->symb);
514 closeItem();
515 break;
516 default:
517 switch(res->type)
518 {
519 case HtmlEntityMapper::Perl_umlaut:
520 accent = "umlaut";
521 break;
522 case HtmlEntityMapper::Perl_acute:
523 accent = "acute";
524 break;
525 case HtmlEntityMapper::Perl_grave:
526 accent = "grave";
527 break;
528 case HtmlEntityMapper::Perl_circ:
529 accent = "circ";
530 break;
531 case HtmlEntityMapper::Perl_slash:
532 accent = "slash";
533 break;
534 case HtmlEntityMapper::Perl_tilde:
535 accent = "tilde";
536 break;
537 case HtmlEntityMapper::Perl_cedilla:
538 accent = "cedilla";
539 break;
540 case HtmlEntityMapper::Perl_ring:
541 accent = "ring";
542 break;
543 default:
544 break;
545 }
546 leaveText();
547 if (accent)
548 {
549 openItem("accent");
550 m_output
551 .addFieldQuotedString("accent", accent)
552 .addFieldQuotedChar("letter", res->symb[0]);
553 closeItem();
554 }
555 break;
556 }
557 }
558 else
559 {
560 err("perl: non supported HTML-entity found: %s\n",HtmlEntityMapper::instance().html(sy.symbol(),TRUE));
561 }
562}
