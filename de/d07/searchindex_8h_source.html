145/** Abstract proxy interface for non-javascript based search indices.
146 * It forwards calls to either SearchIndex or SearchIndexExternal depending
147 * on the Kind passed during construction.
148 */
149class SearchIndexIntf
150{
151 public:
152 using SearchIndexVariant = std::variant<SearchIndex,SearchIndexExternal>;
153 enum Kind { Internal, External };
154 SearchIndexIntf(Kind k) : m_kind(k),
155 m_variant(k==Internal ? SearchIndexVariant(SearchIndex()) :
156 SearchIndexVariant(SearchIndexExternal())) { }
157 void setCurrentDoc(const Definition *ctx,const QCString &anchor,bool isSourceFile)
158 {
159 dispatch_call<SearchIndexMethods::setCurrentDoc>(m_variant,ctx,anchor,isSourceFile);
160 }
161 void addWord(const QCString &word,bool hiPriority)
162 {
163 dispatch_call<SearchIndexMethods::addWord>(m_variant,word,hiPriority);
164 }
165 void write(const QCString &file)
166 {
167 dispatch_call<SearchIndexMethods::write>(m_variant,file);
168 }
169 Kind kind() const { return m_kind; }
170 private:
171 Kind m_kind;
172 SearchIndexVariant m_variant;
173};
