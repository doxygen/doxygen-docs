243/** Class representing a mapping from command names to command IDs. */
244template<typename T>
245class Mapper
246{
247 public:
248 T map(const QCString &n) const
249 {
250 if (n.isEmpty()) return T::UNKNOWN;
251 QCString name = n;
252 if (!m_cs) name=name.lower();
253 auto it = m_map.find(name.str());
254 return it!=m_map.end() ? it->second : T::UNKNOWN;
255 }
256
257 QCString find(const T n) const
258 {
259 for (const auto &[name,id] : m_map)
260 {
261 T curVal = id;
262 // https://stackoverflow.com/a/15889501/1657886
263 if (curVal == n || (curVal == (static_cast<T>(static_cast<int>(n) | static_cast<int>(T::SIMPLESECT_BIT))))) return name.c_str();
264 }
265 return QCString();
266 }
267
268 Mapper(const CommandMap<T> &cm,bool caseSensitive) : m_map(cm), m_cs(caseSensitive)
269 {
270 }
271
272 private:
273 const CommandMap<T> &m_map;
274 bool m_cs;
275};
