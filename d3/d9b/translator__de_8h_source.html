683 /*! used as the title of the HTML page of a class/struct/union */
684 QCString trCompoundReference(const QCString &clName,
685 ClassDef::CompoundType compType,
686 bool isTemplate) override
687 {
688 QCString result=clName+" ";
689 if (isTemplate) result+="Template-";
690 switch(compType)
691 {
692 case ClassDef::Class: result+="Klassen"; break;
693 case ClassDef::Struct: result+="Struktur"; break;
694 case ClassDef::Union: result+="Varianten"; break;
695 case ClassDef::Interface: result+="Schnittstellen"; break;
696 case ClassDef::Protocol: result+="Protokoll"; break;
697 case ClassDef::Category: result+="Kategorie"; break;
698 case ClassDef::Exception: result+="Ausnahmen"; break;
699 default: break;
700 }
701 result+="referenz";
702 return result;
703 }
