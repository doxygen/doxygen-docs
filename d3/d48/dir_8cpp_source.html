280std::string Dir::filePath(const std::string &path,bool acceptsAbsPath) const
281{
282 std::string result;
283 if (acceptsAbsPath && !isRelativePath(path))
284 {
285 result = path;
286 }
287 else
288 {
289 result = (p->path / path).string();
290 }
291 correctPath(result);
292 return result;
293}
