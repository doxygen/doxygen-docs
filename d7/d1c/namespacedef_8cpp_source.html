38static QCString makeDisplayName(const NamespaceDef *nd,bool includeScope)
39{
40 QCString result=includeScope ? nd->name() : nd->localName();
41 SrcLangExt lang = nd->getLanguage();
42 QCString sep = getLanguageSpecificSeparator(lang);
43 if (sep!="::")
44 {
45 result = substitute(result,"::",sep);
46 }
47 //printf("makeDisplayName() %s->%s lang=%d\n",qPrint(name()),qPrint(result),lang);
48 return result;
49}

455void NamespaceDefImpl::insertMember(MemberDef *md)
456{
457 //printf("%s::insertMember(%s) isInline=%d hasDocs=%d\n",qPrint(name()),qPrint(md->name()),
458 // isInline(),hasDocumentation());
459 if (md->isHidden()) return;
460 MemberDefMutable *mdm = toMemberDefMutable(md);
461
462 // if this is an inline namespace that is not documented, then insert the
463 // member in the parent scope instead
464 if (isInline() && !hasDocumentation())
465 {
466 Definition *outerScope = getOuterScope();
467 if (outerScope)
468 {
469 if (outerScope->definitionType()==Definition::TypeNamespace)
470 {
471 NamespaceDefMutable *nd = toNamespaceDefMutable(outerScope);
472 if (nd && nd!=Doxygen::globalScope)
473 {
474 nd->insertMember(md);
475 if (mdm)
476 {
477 mdm->setNamespace(nd);
478 }
479 }
480 }
481 else if (outerScope->definitionType()==Definition::TypeFile)
482 {
483 FileDef *fd = toFileDef(outerScope);
484 fd->insertMember(md);
485 if (mdm)
486 {
487 mdm->setFileDef(fd);
488 mdm->setOuterScope(fd);
489 }
490 }
491 }
492 }
493 else // member is a non-inline namespace or a documented inline namespace
494 {
495 MemberList *allMemberList = getMemberList(MemberListType_allMembersList);
496 if (allMemberList==nullptr)
497 {
498 m_memberLists.emplace_back(std::make_unique<MemberList>(MemberListType_allMembersList,MemberListContainer::Namespace));
499 allMemberList = m_memberLists.back().get();
500 }
501 allMemberList->push_back(md);
502 //printf("%s::m_allMembersDict->append(%s)\n",qPrint(name()),qPrint(md->localName()));
503 m_allMembers.add(md->localName(),md);
504 switch(md->memberType())
505 {
506 case MemberType_Variable:
507 addMemberToList(MemberListType_decVarMembers,md);
508 addMemberToList(MemberListType_docVarMembers,md);
509 break;
510 case MemberType_Function:
511 addMemberToList(MemberListType_decFuncMembers,md);
512 addMemberToList(MemberListType_docFuncMembers,md);
513 break;
514 case MemberType_Typedef:
515 addMemberToList(MemberListType_decTypedefMembers,md);
516 addMemberToList(MemberListType_docTypedefMembers,md);
517 break;
518 case MemberType_Sequence:
519 addMemberToList(MemberListType_decSequenceMembers,md);
520 addMemberToList(MemberListType_docSequenceMembers,md);
521 break;
522 case MemberType_Dictionary:
523 addMemberToList(MemberListType_decDictionaryMembers,md);
524 addMemberToList(MemberListType_docDictionaryMembers,md);
525 break;
526 case MemberType_Enumeration:
527 addMemberToList(MemberListType_decEnumMembers,md);
528 addMemberToList(MemberListType_docEnumMembers,md);
529 break;
530 case MemberType_EnumValue:
531 break;
532 case MemberType_Define:
533 addMemberToList(MemberListType_decDefineMembers,md);
534 addMemberToList(MemberListType_docDefineMembers,md);
535 break;
536 case MemberType_Property:
537 if (md->getLanguage() == SrcLangExt::Python)
538 {
539 addMemberToList(MemberListType_propertyMembers,md);
540 addMemberToList(MemberListType_properties,md);
541 break;
542 }
543 // fallthrough, explicitly no break here
544 default:
545 err("NamespaceDefImpl::insertMembers(): "
546 "member '%s' with unexpected type id %d and class scope '%s' inserted in namespace scope '%s'!\n",
547 qPrint(md->name()),
548 md->memberType(),
549 md->getClassDef() ? qPrint(md->getClassDef()->name()) : "",
550 qPrint(name()));
551 }
552 // if this is an inline namespace, then insert an alias of this member in the outer scope.
553 if (isInline())
554 {
555 Definition *outerScope = getOuterScope();
556 if (outerScope)
557 {
558 std::unique_ptr<MemberDef> aliasMd;
559 if (outerScope->definitionType()==Definition::TypeNamespace)
560 {
561 aliasMd = createMemberDefAlias(outerScope,md);
562 NamespaceDefMutable *ndm = toNamespaceDefMutable(outerScope);
563 if (ndm)
564 {
565 ndm->insertMember(aliasMd.get());
566 }
567 }
568 else if (outerScope->definitionType()==Definition::TypeFile)
569 {
570 aliasMd = createMemberDefAlias(outerScope,md);
571 toFileDef(outerScope)->insertMember(aliasMd.get());
572 }
573 if (aliasMd)
574 {
575 MemberName *mn;
576 QCString name = md->name();
577 mn = Doxygen::functionNameLinkedMap->add(name);
578 mn->push_back(std::move(aliasMd));
579 }
580 }
581 }
582 }
583}

598void NamespaceDefImpl::writeTagFile(TextStream &tagFile)
599{
600 QCString fn = getOutputFileBase();
601 addHtmlExtensionIfMissing(fn);
602 tagFile << " <compound kind=\"namespace\">\n";
603 tagFile << " <name>" << convertToXML(name()) << "</name>\n";
604 tagFile << " <filename>" << fn << "</filename>\n";
605 QCString idStr = id();
606 if (!idStr.isEmpty())
607 {
608 tagFile << " <clangid>" << convertToXML(idStr) << "</clangid>\n";
609 }
610 for (const auto &lde : LayoutDocManager::instance().docEntries(LayoutDocManager::Namespace))
611 {
612 switch (lde->kind())
613 {
614 case LayoutDocEntry::NamespaceNestedNamespaces:
615 {
616 for (const auto &nd : namespaces)
617 {
618 if (nd->isLinkableInProject())
619 {
620 tagFile << " <namespace>" << convertToXML(nd->name()) << "</namespace>\n";
621 }
622 }
623 }
624 break;
625 case LayoutDocEntry::NamespaceClasses:
626 {
627 writeClassesToTagFile(tagFile, classes);
628 }
629 break;
630 case LayoutDocEntry::NamespaceInterfaces:
631 {
632 writeClassesToTagFile(tagFile, interfaces);
633 }
634 break;
635 case LayoutDocEntry::NamespaceStructs:
636 {
637 writeClassesToTagFile(tagFile, structs);
638 }
639 break;
640 case LayoutDocEntry::NamespaceExceptions:
641 {
642 writeClassesToTagFile(tagFile, exceptions);
643 }
644 break;
645 case LayoutDocEntry::NamespaceConcepts:
646 {
647 writeConceptsToTagFile(tagFile);
648 }
649 break;
650 case LayoutDocEntry::MemberDecl:
651 {
652 const LayoutDocEntryMemberDecl *lmd = dynamic_cast<const LayoutDocEntryMemberDecl*>(lde.get());
653 if (lmd)
654 {
655 MemberList * ml = getMemberList(lmd->type);
656 if (ml)
657 {
658 ml->writeTagFile(tagFile);
659 }
660 }
661 }
662 break;
663 case LayoutDocEntry::MemberGroups:
664 {
665 for (const auto &mg : m_memberGroups)
666 {
667 mg->writeTagFile(tagFile);
668 }
669 }
670 break;
671 default:
672 break;
673 }
674 }
675 writeDocAnchorsToTagFile(tagFile);
676 tagFile << " </compound>\n";
677}

679void NamespaceDefImpl::writeDetailedDescription(OutputList &ol,const QCString &title)
680{
681 if (hasDetailedDescription())
682 {
683 ol.pushGeneratorState();
684 ol.disable(OutputType::Html);
685 ol.writeRuler();
686 ol.popGeneratorState();
687 ol.pushGeneratorState();
688 ol.disableAllBut(OutputType::Html);
689 ol.writeAnchor(QCString(),"details");
690 ol.popGeneratorState();
691 ol.startGroupHeader();
692 ol.parseText(title);
693 ol.endGroupHeader();
694
695 ol.startTextBlock();
696 if (!briefDescription().isEmpty() && Config_getBool(REPEAT_BRIEF))
697 {
698 ol.generateDoc(briefFile(),briefLine(),this,nullptr,briefDescription(),FALSE,FALSE,
699 QCString(),FALSE,FALSE,Config_getBool(MARKDOWN_SUPPORT));
700 }
701 if (!briefDescription().isEmpty() && Config_getBool(REPEAT_BRIEF) &&
702 !documentation().isEmpty())
703 {
704 ol.pushGeneratorState();
705 ol.disable(OutputType::Man);
706 ol.disable(OutputType::RTF);
707 ol.enableAll();
708 ol.disableAllBut(OutputType::Man);
709 ol.enable(OutputType::Latex);
710 ol.writeString("\n\n");
711 ol.popGeneratorState();
712 }
713 if (!documentation().isEmpty())
714 {
715 ol.generateDoc(docFile(),docLine(),this,nullptr,documentation()+"\n",TRUE,FALSE,
716 QCString(),FALSE,FALSE,Config_getBool(MARKDOWN_SUPPORT));
717 }
718 ol.endTextBlock();
719 }

1453void NamespaceDefImpl::sortMemberLists()
1454{
1455 for (auto &ml : m_memberLists)
1456 {
1457 if (ml->needsSorting()) { ml->sort(); ml->setNeedsSorting(FALSE); }
1458 }
1459
1460 auto classComp = [](const ClassLinkedRefMap::Ptr &c1,const ClassLinkedRefMap::Ptr &c2)
1461 {
1462 return Config_getBool(SORT_BY_SCOPE_NAME) ?
1463 qstricmp(c1->name(), c2->name())<0 :
1464 qstricmp(c1->className(), c2->className())<0;
1465 };
1466
1467 std::sort(classes.begin(), classes.end(), classComp);
1468 std::sort(interfaces.begin(),interfaces.end(),classComp);
1469 std::sort(structs.begin(), structs.end(), classComp);
1470 std::sort(exceptions.begin(),exceptions.end(),classComp);
1471
1472
1473 auto namespaceComp = [](const NamespaceLinkedRefMap::Ptr &n1,const NamespaceLinkedRefMap::Ptr &n2)
1474 {
1475 return qstricmp(n1->name(),n2->name())<0;
1476 };
1477
1478 std::sort(namespaces.begin(),namespaces.end(),namespaceComp);
