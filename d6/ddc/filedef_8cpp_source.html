56QCString includeStatement(SrcLangExt lang,IncludeKind kind)
57{
58 bool isIDLorJava = lang==SrcLangExt::IDL || lang==SrcLangExt::Java;
59 if (isIDLorJava || (kind & IncludeKind_ImportMask))
60 {
61 return "import ";
62 }
63 else if (kind & IncludeKind_ObjCMask)
64 {
65 return "#import ";
66 }
67 else
68 {
69 return "#include ";
70 }
71}
72
73QCString includeOpen(SrcLangExt lang,IncludeKind kind)
74{
75 if (lang==SrcLangExt::Java || kind==IncludeKind::ImportModule) return "";
76 if (kind & IncludeKind_LocalMask)
77 {
78 return "\"";
79 }
80 else
81 {
82 return "<";
83 }
84}
85
86QCString includeClose(SrcLangExt lang,IncludeKind kind)
87{
88 if (lang==SrcLangExt::Java || lang==SrcLangExt::IDL) return ";";
89 switch (kind)
90 {
91 case IncludeKind::ImportLocal: return "\";";
92 case IncludeKind::ImportLocalObjC: return "\"";
93 case IncludeKind::IncludeLocal: return "\"";
94 case IncludeKind::ImportSystem: return ">;";
95 case IncludeKind::ImportSystemObjC: return ">";
96 case IncludeKind::IncludeSystem: return ">";
97 case IncludeKind::ImportModule: return ";";
98 }
99 return "";
100}
101
102QCString includeTagFileAttributes(SrcLangExt lang,IncludeKind kind)
103{
104 QCString result;
105 bool isIDLorJava = lang==SrcLangExt::IDL || lang==SrcLangExt::Java;
106 result.sprintf("local=\"%s\" import=\"%s\" module=\"%s\" objc=\"%s\"",
107 (kind & IncludeKind_LocalMask) ? "yes" : "no",
108 (isIDLorJava || (kind & IncludeKind_ImportMask)) ? "yes" : "no",
109 (kind==IncludeKind::ImportModule) ? "yes" : "no",
110 (kind & IncludeKind_ObjCMask) ? "yes" : "no");
111 return result;
112}

371void FileDefImpl::writeTagFile(TextStream &tagFile)
372{
373 QCString fn=getOutputFileBase();
374 addHtmlExtensionIfMissing(fn);
375 tagFile << " <compound kind=\"file\">\n";
376 tagFile << " <name>" << convertToXML(name()) << "</name>\n";
377 tagFile << " <path>" << convertToXML(stripFromPath(getPath())) << "</path>\n";
378 tagFile << " <filename>" << fn << "</filename>\n";
379 for (const auto &ii : m_includeList)
380 {
381 const FileDef *fd=ii.fileDef;
382 if (fd && fd->isLinkable() && !fd->isReference())
383 {
384 QCString attr = includeTagFileAttributes(fd->getLanguage(),ii.kind);
385 tagFile << " <includes id=\""
386 << convertToXML(fd->getOutputFileBase()) << "\" "
387 << "name=\"" << convertToXML(fd->name()) << "\" "
388 << attr << ">"
389 << convertToXML(ii.includeName)
390 << "</includes>\n";
391 }
392 }
393 for (const auto &lde : LayoutDocManager::instance().docEntries(LayoutDocManager::File))
394 {
395 switch (lde->kind())
396 {
397 case LayoutDocEntry::FileClasses:
398 {
399 writeClassesToTagFile(tagFile, m_classes);
400 }
401 break;
402 case LayoutDocEntry::FileInterfaces:
403 {
404 writeClassesToTagFile(tagFile, m_interfaces);
405 }
406 break;
407 case LayoutDocEntry::FileStructs:
408 {
409 writeClassesToTagFile(tagFile, m_structs);
410 }
411 break;
412 case LayoutDocEntry::FileExceptions:
413 {
414 writeClassesToTagFile(tagFile, m_exceptions);
415 }
416 break;
417 case LayoutDocEntry::FileConcepts:
418 {
419 for (const auto *nd : m_concepts)
420 {
421 if (nd->isLinkableInProject())
422 {
423 tagFile << " <concept>" << convertToXML(nd->name()) << "</concept>\n";
424 }
425 }
426 }
427 break;
428 case LayoutDocEntry::FileNamespaces:
429 {
430 for (const auto *nd : m_namespaces)
431 {
432 if (nd->isLinkableInProject())
433 {
434 tagFile << " <namespace>" << convertToXML(nd->name()) << "</namespace>\n";
435 }
436 }
437 }
438 break;
439 case LayoutDocEntry::MemberDecl:
440 {
441 const LayoutDocEntryMemberDecl *lmd = dynamic_cast<const LayoutDocEntryMemberDecl*>(lde.get());
442 if (lmd)
443 {
444 MemberList * ml = getMemberList(lmd->type);
445 if (ml)
446 {
447 ml->writeTagFile(tagFile,false,false);
448 }
449 }
450 }
451 break;
452 case LayoutDocEntry::MemberGroups:
453 {
454 for (const auto &mg : m_memberGroups)
455 {
456 mg->writeTagFile(tagFile);
457 }
458 }
459 break;
460 default:
461 break;
462 }
463 }
464
465 writeDocAnchorsToTagFile(tagFile);
466 tagFile << " </compound>\n";
467}

469void FileDefImpl::writeDetailedDescription(OutputList &ol,const QCString &title)
470{
471 if (hasDetailedDescription())
472 {
473 ol.pushGeneratorState();
474 ol.disable(OutputType::Html);
475 ol.writeRuler();
476 ol.popGeneratorState();
477 ol.pushGeneratorState();
478 ol.disableAllBut(OutputType::Html);
479 ol.writeAnchor(QCString(),"details");
480 ol.popGeneratorState();
481 ol.startGroupHeader();
482 ol.parseText(title);
483 ol.endGroupHeader();
484
485 ol.startTextBlock();
486 if (!briefDescription().isEmpty() && Config_getBool(REPEAT_BRIEF))
487 {
488 ol.generateDoc(briefFile(),briefLine(),this,nullptr,briefDescription(),FALSE,FALSE,
489 QCString(),FALSE,FALSE,Config_getBool(MARKDOWN_SUPPORT));
490 }
491 if (!briefDescription().isEmpty() && Config_getBool(REPEAT_BRIEF) &&
492 !documentation().isEmpty())
493 {
494 ol.pushGeneratorState();
495 ol.disable(OutputType::Man);
496 ol.disable(OutputType::RTF);
497 ol.enableAll();
498 ol.disableAllBut(OutputType::Man);
499 ol.enable(OutputType::Latex);
500 ol.writeString("\n\n");
501 ol.popGeneratorState();
502 }
503 if (!documentation().isEmpty())
504 {
505 ol.generateDoc(docFile(),docLine(),this,nullptr,documentation()+"\n",TRUE,FALSE,
506 QCString(),FALSE,FALSE,Config_getBool(MARKDOWN_SUPPORT));
507 }
508 //printf("Writing source ref for file %s\n",qPrint(name()));
509 if (Config_getBool(SOURCE_BROWSER))
510 {
511 ol.startParagraph("definition");
512 QCString refText = theTranslator->trDefinedInSourceFile();
513 int fileMarkerPos = refText.find("@0");
514 if (fileMarkerPos!=-1) // should always pass this.
515 {
516 ol.parseText(refText.left(fileMarkerPos)); //text left from marker 1
517 ol.writeObjectLink(QCString(),getSourceFileBase(),QCString(),name());
518 ol.parseText(refText.right(
519 refText.length()-fileMarkerPos-2)); // text right from marker 2
520 }
521 else
522 {
523 err("translation error: invalid marker in trDefinedInSourceFile()\n");
524 }
525 ol.endParagraph();
526 }
527 ol.endTextBlock();
528 }
529}

581void FileDefImpl::writeIncludeFiles(OutputList &ol)
582{
583 if (!m_includeList.empty())
584 {
585 ol.startTextBlock(TRUE);
586 for (const auto &ii : m_includeList)
587 {
588 const FileDef *fd=ii.fileDef;
589 ol.startTypewriter();
590 SrcLangExt lang = fd ? fd->getLanguage() : SrcLangExt::Cpp;
591 ol.docify(::includeStatement(lang,ii.kind));
592 ol.docify(::includeOpen(lang,ii.kind));
593 ol.disable(OutputType::Html);
594 ol.docify(ii.includeName);
595 ol.enableAll();
596 ol.disableAllBut(OutputType::Html);
597
598 // Here we use the include file name as it appears in the file.
599 // we could also we the name as it is used within doxygen,
600 // then we should have used fd->docName() instead of ii->includeName
601 if (fd && fd->isLinkable())
602 {
603 ol.writeObjectLink(fd->getReference(),
604 fd->generateSourceFile() ? fd->includeName() : fd->getOutputFileBase(),
605 QCString(),ii.includeName);
606 }
607 else
608 {
609 ModuleDef *mod = ModuleManager::instance().getPrimaryInterface(ii.includeName);
610 if (ii.kind==IncludeKind::ImportModule && mod && mod->isLinkable())
611 {
612 ol.writeObjectLink(mod->getReference(),mod->getOutputFileBase(),
613 QCString(),ii.includeName);
614 }
615 else
616 {
617 ol.docify(ii.includeName);
618 }
619 }
620
621 ol.enableAll();
622 ol.docify(::includeClose(lang,ii.kind));
623 ol.endTypewriter();
624 ol.lineBreak();
625 }
626 ol.endTextBlock();
627 }
