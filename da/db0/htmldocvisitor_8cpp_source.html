78static QCString convertIndexWordToAnchor(const QCString &word)
79{
80 static int cnt = 0;
81 QCString result="a";
82 QCString cntStr;
83 result += cntStr.setNum(cnt);
84 result += "_";
85 cnt++;
86 const char *str = word.data();
87 unsigned char c = 0;
88 if (str)
89 {
90 while ((c = *str++))
91 {
92 if ((c >= 'a' && c <= 'z') || // ALPHA
93 (c >= 'A' && c <= 'Z') || // ALPHA
94 (c >= '0' && c <= '9') || // DIGIT
95 c == '-' ||
96 c == '.'
97 )
98 {
99 result += c;
100 }
101 else
102 {
103 char enc[4];
104 enc[0] = '_';
105 enc[1] = hex[(c & 0xf0) >> 4];
106 enc[2] = hex[c & 0xf];
107 enc[3] = 0;
108 result += enc;
109 }
110 }
111 }
112 return result;
113}

117static bool mustBeOutsideParagraph(const DocNodeVariant &n)
118{
119 //printf("mustBeOutsideParagraph(%s)=",docNodeName(n));
120 if (holds_one_of_alternatives< /* <ul> */ DocHtmlList, DocSimpleList, DocAutoList,
121 /* <dl> */ DocSimpleSect, DocParamSect, DocHtmlDescList, DocXRefItem,
122 /* <table> */ DocHtmlTable,
123 /* <h?> */ DocSection, DocHtmlHeader,
124 /* \internal */ DocInternal,
125 /* <div> */ DocInclude, DocSecRefList,
126 /* <hr> */ DocHorRuler,
127 /* <blockquote> */ DocHtmlBlockQuote,
128 /* \parblock */ DocParBlock,
129 /* \dotfile */ DocDotFile,
130 /* \mscfile */ DocMscFile,
131 /* \diafile */ DocDiaFile,
132 /* \plantumlfile */ DocPlantUmlFile,
133 /* <details> */ DocHtmlDetails,
134 /* <summary> */ DocHtmlSummary,
135 DocIncOperator >(n))
136 {
137 return TRUE;
138 }
139 const DocVerbatim *dv = std::get_if<DocVerbatim>(&n);
140 if (dv)
141 {
142 DocVerbatim::Type t = dv->type();
143 if (t == DocVerbatim::JavaDocCode || t == DocVerbatim::JavaDocLiteral) return FALSE;
144 return t!=DocVerbatim::HtmlOnly || dv->isBlock();
145 }
146 const DocStyleChange *sc = std::get_if<DocStyleChange>(&n);
147 if (sc)
148 {
149 return sc->style()==DocStyleChange::Preformatted ||
150 sc->style()==DocStyleChange::Div ||
151 sc->style()==DocStyleChange::Center;
152 }
153 const DocFormula *df = std::get_if<DocFormula>(&n);
154 if (df)
155 {
156 return !df->isInline();
157 }
158 const DocImage *di = std::get_if<DocImage>(&n);
159 if (di)
160 {
161 return !di->isInlineImage();
162 }
163 return FALSE;
164}

208static bool isInvisibleNode(const DocNodeVariant &node)
209{
210 //printf("isInvisibleNode(%s)\n",docNodeName(node));
211 // skip over white space
212 const DocWhiteSpace *ws = std::get_if<DocWhiteSpace>(&node);
213 if (ws) return true;
214 // skip over image nodes that are not for HTML output
215 const DocImage *di = std::get_if<DocImage>(&node);
216 if (di) return di->type()!=DocImage::Html;
217 // skip over verbatim nodes that are not visible in the HTML output
218 const DocVerbatim *dv = std::get_if<DocVerbatim>(&node);
219 if (dv) return !isDocVerbatimVisible(*dv);
220 // skip over include nodes that are not visible in the HTML output
221 const DocInclude *dinc = std::get_if<DocInclude>(&node);
222 if (dinc) return !isDocIncludeVisible(*dinc);
223 const DocIncOperator *dio = std::get_if<DocIncOperator>(&node);
224 // skip over include operator nodes that are not visible in the HTML output
225 if (dio) return !isDocIncOperatorVisible(*dio);
226 return false;
227}

229static void mergeHtmlAttributes(const HtmlAttribList &attribs, HtmlAttribList &mergeInto)
230{
231 for (const auto &att : attribs)
232 {
233 auto it = std::find_if(std::begin(mergeInto),std::end(mergeInto),
234 [&att](const auto &opt) { return opt.name==att.name; });
235 if (it!=std::end(mergeInto)) // attribute name already in mergeInto
236 {
237 it->value = it->value + " " + att.value;
238 }
239 else // attribute name not yet in mergeInto
240 {
241 mergeInto.push_back(att);
242 }
243 }
244}

246static QCString htmlAttribsToString(const HtmlAttribList &attribs, QCString *pAltValue = nullptr)
247{
248 QCString result;
249 for (const auto &att : attribs)
250 {
251 if (!att.value.isEmpty()) // ignore attribute without values as they
252 // are not XHTML compliant, with the exception
253 // of the alt attribute with the img tag
254 {
255 if (att.name=="alt" && pAltValue) // optionally return the value of alt separately
256 // need to convert <img> to <object> for SVG images,
257 // which do not support the alt attribute
258 {
259 *pAltValue = att.value;
260 }
261 else
262 {
263 result+=" ";
264 result+=att.name;
265 result+="=\""+convertToXML(att.value)+"\"";
266 }
267 }
268 else if (att.name=="open")
269 {
270 // The open attribute is a boolean attribute.
271 // Specifies that the details should be visible (open) to the user
272 // As it is a boolean attribute the initialization value is of no interest
273 result+=" ";
274 result+=att.name;
275 result+="=\"true\"";
276 }
277 else if (att.name=="nowrap") // In XHTML, attribute minimization is forbidden, and the nowrap attribute must be defined as <td nowrap="nowrap">.
278 {
279 result+=" ";
280 result+=att.name;
281 result+="=\"nowrap\"";
282 }
283 }
284 return result;
285}

569void HtmlDocVisitor::operator()(const DocVerbatim &s)
570{
571 if (m_hide) return;
572 QCString lang = m_langExt;
573 if (!s.language().isEmpty()) // explicit language setting
574 {
575 lang = s.language();
576 }
577 SrcLangExt langExt = getLanguageFromCodeLang(lang);
578 switch(s.type())
579 {
580 case DocVerbatim::Code:
581 forceEndParagraph(s);
582 m_ci.startCodeFragment("DoxyCode");
583 getCodeParser(lang).parseCode(m_ci,
584 s.context(),
585 s.text(),
586 langExt,
587 Config_getBool(STRIP_CODE_COMMENTS),
588 s.isExample(),
589 s.exampleFile(),
590 nullptr, // fileDef
591 -1, // startLine
592 -1, // endLine
593 true, // inlineFragment
594 nullptr, // memberDef
595 true, // show line numbers
596 m_ctx // search context
597 );
598 m_ci.endCodeFragment("DoxyCode");
599 forceStartParagraph(s);
600 break;
601 case DocVerbatim::Verbatim:
602 forceEndParagraph(s);
603 m_t << "<pre class=\"fragment\">";
604 filter(s.text());
605 m_t << "</pre>";
606 forceStartParagraph(s);
607 break;
608 case DocVerbatim::JavaDocLiteral:
609 filter(s.text(), true);
610 break;
611 case DocVerbatim::JavaDocCode:
612 m_t << "<code class=\"JavaDocCode\">";
613 filter(s.text(), true);
614 m_t << "</code>";
615 break;
616 case DocVerbatim::HtmlOnly:
617 {
618 if (s.isBlock()) forceEndParagraph(s);
619 m_t << s.text();
620 if (s.isBlock()) forceStartParagraph(s);
621 }
622 break;
623 case DocVerbatim::ManOnly:
624 case DocVerbatim::LatexOnly:
625 case DocVerbatim::XmlOnly:
626 case DocVerbatim::RtfOnly:
627 case DocVerbatim::DocbookOnly:
628 /* nothing */
629 break;
630
631 case DocVerbatim::Dot:
632 {
633 static int dotindex = 1;
634 QCString fileName(4096, QCString::ExplicitSize);
635
636 forceEndParagraph(s);
637 fileName.sprintf("%s%d%s",
638 qPrint(Config_getString(HTML_OUTPUT)+"/inline_dotgraph_"),
639 dotindex++,
640 ".dot"
641 );
642 std::ofstream file = Portable::openOutputStream(fileName);
643 if (!file.is_open())
644 {
645 err("Could not open file %s for writing\n",qPrint(fileName));
646 }
647 else
648 {
649 QCString stext = s.text();
650 file.write( stext.data(), stext.length() );
651 file.close();
652
653 m_t << "<div class=\"dotgraph\">\n";
654 writeDotFile(fileName,s.relPath(),s.context(),s.srcFile(),s.srcLine());
655 visitCaption(m_t, s);
656 m_t << "</div>\n";
657
658 if (Config_getBool(DOT_CLEANUP)) Dir().remove(fileName.str());
659 }
660 forceStartParagraph(s);
661 }
662 break;
663 case DocVerbatim::Msc:
664 {
665 forceEndParagraph(s);
666
667 static int mscindex = 1;
668 QCString baseName(4096, QCString::ExplicitSize);
669
670 baseName.sprintf("%s%d",
671 qPrint(Config_getString(HTML_OUTPUT)+"/inline_mscgraph_"),
672 mscindex++
673 );
674 std::ofstream file = Portable::openOutputStream(baseName.str()+".msc");
675 if (!file.is_open())
676 {
677 err("Could not open file %s.msc for writing\n",qPrint(baseName));
678 }
679 else
680 {
681 QCString text = "msc {";
682 text+=s.text();
683 text+="}";
684
685 file.write( text.data(), text.length() );
686 file.close();
687
688 m_t << "<div class=\"mscgraph\">\n";
689 writeMscFile(baseName+".msc",s.relPath(),s.context(),s.srcFile(),s.srcLine());
690 visitCaption(m_t, s);
691 m_t << "</div>\n";
692
693 if (Config_getBool(DOT_CLEANUP)) Dir().remove(baseName.str()+".msc");
694 }
695 forceStartParagraph(s);
696 }
697 break;
698 case DocVerbatim::PlantUML:
699 {
700 forceEndParagraph(s);
701 QCString htmlOutput = Config_getString(HTML_OUTPUT);
702 QCString imgExt = getDotImageExtension();
703 PlantumlManager::OutputFormat format = PlantumlManager::PUML_BITMAP; // default : PUML_BITMAP
704 if (imgExt=="svg")
705 {
706 format = PlantumlManager::PUML_SVG;
707 }
708 QCString baseName = PlantumlManager::instance().writePlantUMLSource(
709 htmlOutput,s.exampleFile(),
710 s.text(),format,s.engine(),s.srcFile(),s.srcLine(),true);
711 m_t << "<div class=\"plantumlgraph\">\n";
712 writePlantUMLFile(baseName,s.relPath(),s.context(),s.srcFile(),s.srcLine());
713 visitCaption(m_t, s);
714 m_t << "</div>\n";
715 forceStartParagraph(s);
716 }
717 break;
718 }

757 m_ci.startCodeFragment("DoxyCode");

776 forceStartParagraph(inc);
777 }

1126{
1127 const DocNodeList &nodes = parent.children();
1128 auto it = std::find_if(std::begin(nodes),std::end(nodes),[&par](const auto &n) { return holds_value(&par,n); });
1129 if (it==std::end(nodes)) return FALSE;
1130 size_t count = parent.children().size();

1132 if (count>1 && it==std::begin(nodes)) // it points to first node
1133 {
1134 return isSeparator(std::next(it));

1136 else if (count>1 && it==std::prev(std::end(nodes))) // it points to last node
1137 {
1138 return isSeparator(std::prev(it));
1139 }
1140 else if (count>2 && it!=std::begin(nodes) && it!=std::prev(std::end(nodes))) // it points to intermediate node
1141 {
1142 return isSeparator(std::prev(it)) && isSeparator(std::next(it));
1143 }

1147static contexts_t getParagraphContext(const DocPara &p,bool &isFirst,bool &isLast)
1148{
1149 contexts_t t=contexts_t::NONE;
1150 isFirst=FALSE;
1151 isLast=FALSE;
1152 if (p.parent())
1153 {
1154 const auto parBlock = std::get_if<DocParBlock>(p.parent());
1155 if (parBlock)
1156 {
1157 // hierarchy: node N -> para -> parblock -> para
1158 // adapt return value to kind of N
1159 const DocNodeVariant *p3 = nullptr;
1160 if (::parent(p.parent()) && ::parent(::parent(p.parent())) )
1161 {
1162 p3 = ::parent(::parent(p.parent()));

1165 isLast =isLastChildNode (parBlock,p);

1167 bool isDD = p3!=nullptr && holds_one_of_alternatives<DocHtmlDescData,DocXRefItem,DocSimpleSect>(*p3);

1169 t=contexts_t::NONE;

1171 {
1172 if (isLI) t=contexts_t::STARTLI; else if (isDD) t=contexts_t::STARTDD; else if (isTD) t=contexts_t::STARTTD;

1174 if (isLast)
1175 {
1176 if (isLI) t=contexts_t::ENDLI; else if (isDD) t=contexts_t::ENDDD; else if (isTD) t=contexts_t::ENDTD;
1177 }
1178 if (!isFirst && !isLast)
1179 {
1180 if (isLI) t=contexts_t::INTERLI; else if (isDD) t=contexts_t::INTERDD; else if (isTD) t=contexts_t::INTERTD;

1183 }

1191 }

1193 if (docSimpleListItem)

1234 if (isLast) t=contexts_t::ENDDD;
1235 if (!isFirst && !isLast) t = contexts_t::INTERDD;

1237 }

1239 if (docXRefItem)

1309void HtmlDocVisitor::operator()(const DocPara &p)
1310{
1311 if (m_hide) return;

1313 //printf("> DocPara\n");
1314 //dumpDocNodeList(p.children());
1315
1316 bool needsTag = determineIfNeedsTag(p);
1317 //printf(" needsTag=%d\n",needsTag);
1318 bool needsTagBefore = needsTag;
1319 bool needsTagAfter = needsTag;
1320
1321 // if the first element of a paragraph is something that should be outside of
1322 // the paragraph (<ul>,<dl>,<table>,..) then that will already started the
1323 // paragraph and we don't need to do it here
1324 if (!p.children().empty())
1325 {
1326 auto it = std::find_if(std::begin(p.children()),std::end(p.children()),
1327 [](const auto &node) { return !isInvisibleNode(node); });
1328 if (it!=std::end(p.children()))
1329 {
1330 const DocNodeVariant &n = *it;
1331 if (mustBeOutsideParagraph(n))
1332 {
1333 needsTagBefore = FALSE;
1334 }
1335 }
1336 }
1337
1338 // check if this paragraph is the first or last or intermediate child of a <li> or <dd>.
1339 // this allows us to mark the tag with a special class so we can
1340 // fix the otherwise ugly spacing.
1341 bool isFirst = false;
1342 bool isLast = false;
1343 contexts_t t = getParagraphContext(p,isFirst,isLast);
1344 //printf("startPara first=%d last=%d\n",isFirst,isLast);
1345 if (isFirst && isLast) needsTagBefore=FALSE;
1346
1347 //printf(" needsTagBefore=%d\n",needsTagBefore);
1348 // write the paragraph tag (if needed)
1349 if (needsTagBefore)
1350 {
1351 if (contexts(t))
1352 m_t << "<p class=\"" << contexts(t) << "\"" << htmlAttribsToString(p.attribs()) << ">";
1353 else
1354 m_t << "<p" << htmlAttribsToString(p.attribs()) << ">";
1355 }
1356
1357 visitChildren(p);
1358
1359 // if the last element of a paragraph is something that should be outside of
1360 // the paragraph (<ul>,<dl>,<table>) then that will already have ended the
1361 // paragraph and we don't need to do it here
1362 if (!p.children().empty())
1363 {
1364 auto it = std::prev(std::end(p.children()));
1365 for (;;)
1366 {
1367 const DocNodeVariant &n = *it;
1368 if (!isInvisibleNode(n))
1369 {
1370 if (mustBeOutsideParagraph(n))
1371 {
1372 needsTagAfter = FALSE;
1373 }
1374 // stop searching if we found a node that is visible
1375 break;
1376 }
1377 if (it==std::begin(p.children()))
1378 {
1379 // stop searching if we are at the beginning of the list
1380 break;
1381 }
1382 else
1383 {
1384 --it;
1385 }
1386 }
1387 }
1388
1389 //printf("endPara first=%d last=%d\n",isFirst,isLast);
1390 if (isFirst && isLast) needsTagAfter=FALSE;
1391
1392 //printf(" needsTagAfter=%d\n",needsTagAfter);
1393 if (needsTagAfter) m_t << "</p>\n";
1394 //printf("< DocPara\n");
1395}

2126{

2135 case '<': m_t << "&lt;"; break;
2136 case '>': m_t << "&gt;"; break;
2137 case '&': m_t << "&amp;"; break;
2138 case '\\': if ((*p == '(') || (*p == ')'))
2139 m_t << "\\&zwj;" << *p++;
2140 else
2141 m_t << c;
2142 break;
2143 default:

2145 uint8_t uc = static_cast<uint8_t>(c);
2146 if (uc<32 && !isspace(c)) // non-printable control characters
