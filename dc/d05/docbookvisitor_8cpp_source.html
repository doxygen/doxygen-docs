51static QCString filterId(const QCString &s)
52{
53 if (s.isEmpty()) return s;
54 GrowBuf growBuf;
55 growBuf.clear();
56 const char *p=s.data();
57 char c=0;
58 while ((c=*p++))
59 {
60 switch (c)
61 {
62 case ':': growBuf.addStr("_1"); break;
63 default: growBuf.addChar(c); break;
64 }
65 }
66 growBuf.addChar(0);
67 return growBuf.get();
68}

88 QCString result = baseName;

117void DocbookDocVisitor::visitPreStart(TextStream &t,

119 bool hasCaption,

121 const QCString &width,
122 const QCString &height,
123 bool inlineImage)
124{
125 if (hasCaption && !inlineImage)

130 t << " </title>\n";
131 }
132 else
133 {
134 t << " <informalfigure>\n";
135 }
136 t << " <mediaobject>\n";
137 t << " <imageobject>\n";
138 t << " <imagedata";
139 if (!width.isEmpty())
140 {
141 t << " width=\"" << convertToDocBook(width) << "\"";
142 }
143 else
144 {
145 if (!height.isEmpty() && !inlineImage) t << " width=\"50%\"";
146 }
147 if (!height.isEmpty())
148 {
149 t << " depth=\"" << convertToDocBook(height) << "\"";
150 }
151 t << " align=\"center\" valign=\"middle\" scalefit=\"0\" fileref=\"" << name << "\">";
152 t << "</imagedata>\n";
153 t << " </imageobject>\n";
154 if (hasCaption && !inlineImage)
155 {
156 t << " <!--\n"; // Needed for general formatting with title for other formats
157 }
158}

329void DocbookDocVisitor::operator()(const DocVerbatim &s)
330{
331DB_VIS_C
332 if (m_hide) return;
333 QCString lang = m_langExt;
334 if (!s.language().isEmpty()) // explicit language setting
335 {
336 lang = s.language();
337 }
338 SrcLangExt langExt = getLanguageFromCodeLang(lang);
339 switch(s.type())
340 {
341 case DocVerbatim::Code:
342 m_t << "<literallayout><computeroutput>";
343 getCodeParser(m_langExt).parseCode(m_ci,s.context(),
344 s.text(),
345 langExt,
346 Config_getBool(STRIP_CODE_COMMENTS),
347 s.isExample(),
348 s.exampleFile());
349 m_t << "</computeroutput></literallayout>";
350 break;
351 case DocVerbatim::Verbatim:
352 m_t << "<literallayout><computeroutput>";
353 filter(s.text());
354 m_t << "</computeroutput></literallayout>";
355 break;
356 case DocVerbatim::JavaDocLiteral:
357 filter(s.text(), true);
358 break;
359 case DocVerbatim::JavaDocCode:
360 m_t << "<computeroutput>";
361 filter(s.text(), true);
362 m_t << "</computeroutput>";
363 break;
364 case DocVerbatim::HtmlOnly:
365 break;
366 case DocVerbatim::RtfOnly:
367 break;
368 case DocVerbatim::ManOnly:
369 break;
370 case DocVerbatim::LatexOnly:
371 break;
372 case DocVerbatim::XmlOnly:
373 break;
374 case DocVerbatim::DocbookOnly:
375 m_t << s.text();
376 break;
377 case DocVerbatim::Dot:
378 {
379 static int dotindex = 1;
380 QCString baseName(4096, QCString::ExplicitSize);
381 QCString name;
382 QCString stext = s.text();
383 m_t << "<para>\n";
384 name.sprintf("%s%d", "dot_inline_dotgraph_", dotindex);
385 baseName.sprintf("%s%d",
386 qPrint(Config_getString(DOCBOOK_OUTPUT)+"/inline_dotgraph_"),
387 dotindex++
388 );
389 QCString fileName = baseName+".dot";
390 std::ofstream file = Portable::openOutputStream(fileName);
391 if (!file.is_open())
392 {
393 err("Could not open file %s for writing\n",qPrint(fileName));
394 }
395 file.write( stext.data(), stext.length() );
396 file.close();
397 writeDotFile(baseName, s);
398 m_t << "</para>\n";
399 if (Config_getBool(DOT_CLEANUP)) Dir().remove(fileName.str());
400 }
401 break;
402 case DocVerbatim::Msc:
403 {
404 static int mscindex = 1;
405 QCString baseName(4096, QCString::ExplicitSize);
406 QCString name;
407 QCString stext = s.text();
408 m_t << "<para>\n";
409 name.sprintf("%s%d", "msc_inline_mscgraph_", mscindex);
410 baseName.sprintf("%s%d",
411 (Config_getString(DOCBOOK_OUTPUT)+"/inline_mscgraph_").data(),
412 mscindex++
413 );
414 QCString fileName = baseName+".msc";
415 std::ofstream file = Portable::openOutputStream(fileName);
416 if (!file.is_open())
417 {
418 err("Could not open file %s for writing\n",qPrint(fileName));
419 }
420 QCString text = "msc {";
421 text+=stext;
422 text+="}";
423 file.write( text.data(), text.length() );
424 file.close();
425 writeMscFile(baseName,s);
426 m_t << "</para>\n";
427 if (Config_getBool(DOT_CLEANUP)) Dir().remove(fileName.str());
428 }
429 break;
430 case DocVerbatim::PlantUML:
431 {
432 QCString docbookOutput = Config_getString(DOCBOOK_OUTPUT);
433 QCString baseName = PlantumlManager::instance().writePlantUMLSource(docbookOutput,
434 s.exampleFile(),s.text(),PlantumlManager::PUML_BITMAP,
435 s.engine(),s.srcFile(),s.srcLine(),true);
436 QCString shortName = makeShortName(baseName);
437 m_t << "<para>\n";
438 writePlantUMLFile(baseName,s);
439 m_t << "</para>\n";
440 }
441 break;
442 }
443}

515void DocbookDocVisitor::operator()(const DocIncOperator &op)
516{
517DB_VIS_C
518 if (op.isFirst())
519 {
520 if (!m_hide)
521 {
522 m_t << "<programlisting linenumbering=\"unnumbered\">";
523 }
524 pushHidden(m_hide);
525 m_hide = TRUE;
526 }
527 QCString locLangExt = getFileNameExtension(op.includeFileName());
528 if (locLangExt.isEmpty()) locLangExt = m_langExt;
529 SrcLangExt langExt = getLanguageFromFileName(locLangExt);
530 if (op.type()!=DocIncOperator::Skip)
531 {
532 m_hide = popHidden();
533 if (!m_hide)
534 {
535 std::unique_ptr<FileDef> fd;
536 if (!op.includeFileName().isEmpty())
537 {
538 FileInfo cfi( op.includeFileName().str() );
539 fd = createFileDef( cfi.dirPath(), cfi.fileName() );
540 }
541
542 getCodeParser(locLangExt).parseCode(m_ci,op.context(),
543 op.text(),langExt,
544 op.stripCodeComments(),
545 op.isExample(),
546 op.exampleFile(),
547 fd.get(), // fileDef
548 op.line(), // startLine
549 -1, // endLine
550 FALSE, // inline fragment
551 nullptr, // memberDef
552 op.showLineNo() // show line numbers
553 );
554 }
555 pushHidden(m_hide);
556 m_hide=TRUE;
557 }
558 if (op.isLast())
559 {
560 m_hide = popHidden();
561 if (!m_hide) m_t << "</programlisting>";
562 }
563 else
564 {
565 if (!m_hide) m_t << "\n";
566 }
567}

871 break;

874 break;
