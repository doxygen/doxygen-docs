135bool OutputList::isEnabled(OutputType o)
136{
137 bool found = false;
138 bool enabled = false;
139 for (const auto &e : m_outputGenList)
140 {
141 std::visit([&](auto &&gen) {
142 if (gen.type()==o) { enabled = e.enabled; found=true; }
143 }, e.variant);
144 if (found) return enabled;
145 }
146 return enabled;
147}

159void OutputList::popGeneratorState()
160{
161 //printf("popGeneratorState()\n");
162 for (auto &e : m_outputGenList)
163 {
164 if (!e.enabledStack.empty())
165 {
166 e.enabled = e.enabledStack.top();
167 e.enabledStack.pop();
168 }
169 }
170 syncEnabled();
171}

173void OutputList::generateDoc(const QCString &fileName,int startLine,
174 const Definition *ctx,const MemberDef * md,
175 const QCString &docStr,bool indexWords,
176 bool isExample,const QCString &exampleName,
177 bool singleLine,bool linkFromIndex,
178 bool markdownSupport)
179{
180 if (docStr.isEmpty()) return;
181
182 auto count=std::count_if(m_outputGenList.begin(),m_outputGenList.end(),
183 [](const auto &e) { return e.enabled; });
184 if (count>0)
185 {
186 // we want to validate irrespective of the number of output formats
187 // specified as:
188 // - when only XML format there should be warnings as well (XML has its own write routines)
189 // - no formats there should be warnings as well
190 auto parser { createDocParser() };
191 auto ast { validatingParseDoc(*parser.get(),
192 fileName,startLine,
193 ctx,md,docStr,indexWords,isExample,exampleName,
194 singleLine,linkFromIndex,markdownSupport) };
195 if (ast) writeDoc(ast.get(),ctx,md);
196 }
197}

206void OutputList::parseText(const QCString &textStr)
207{
208
209 auto count=std::count_if(m_outputGenList.begin(),m_outputGenList.end(),
210 [](const auto &e) { return e.enabled; });
211
212 if (count>0)
213 {
214 // we want to validate irrespective of the number of output formats
215 // specified as:
216 // - when only XML format there should be warnings as well (XML has its own write routines)
217 // - no formats there should be warnings as well

220
221 if (ast) writeDoc(ast.get(),nullptr,nullptr);

227void OutputCodeRecorder::startNewLine(int lineNr)
228{
229 int orgSize = static_cast<int>(m_lineOffset.size());
230 if (orgSize<lineNr)
231 {
232 m_lineOffset.resize(lineNr);
233 for (int i=orgSize;i<lineNr;i++) // output lines can be skipped due to hidden comments so fill in the gap
234 {
235 //printf("%p: startCodeLine(%d) offset=%zu\n",(void*)this,i,m_calls.size());
236 m_lineOffset[i]=m_calls.size();
237 }
238 }

333void OutputCodeRecorder::replay(OutputCodeList &ol,int startLine,int endLine,bool showLineNumbers)
334{
335 size_t startIndex = startLine>0 && startLine<=(int)m_lineOffset.size() ? m_lineOffset[startLine-1] : 0;
336 size_t endIndex = endLine>0 && endLine <=(int)m_lineOffset.size() ? m_lineOffset[ endLine-1] : m_calls.size();
337 //printf("startIndex=%zu endIndex=%zu\n",startIndex,endIndex);
338 m_showLineNumbers = showLineNumbers;
339 for (size_t i=startIndex; i<endIndex; i++)
340 {
341 if (m_calls[i].condition()) m_calls[i].function(&ol);
342 }
