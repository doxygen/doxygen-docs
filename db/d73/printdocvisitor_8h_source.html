733 // helper functions
734 void indent()
735 {
736 if (m_needsEnter) printf("\n");
737 for (int i=0;i<m_indent;i++) printf(".");
738 m_needsEnter=FALSE;
739 }
740 void indent_leaf()
741 {
742 if (!m_needsEnter) indent();
743 m_needsEnter=TRUE;
744 }
745 void indent_pre()
746 {
747 indent();
748 m_indent++;
749 }
750 void indent_post()
751 {
752 m_indent--;
753 indent();
754 }
