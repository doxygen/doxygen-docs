93static void visitPreStart(TextStream &t, const char *cmd, bool doCaption,
94 XmlDocVisitor &visitor, const DocNodeList &children,
95 const QCString &name, bool writeType, DocImage::Type type, const QCString &width,
96 const QCString &height, const QCString engine = QCString(), const QCString &alt = QCString(), bool inlineImage = FALSE)
97{
98 t << "<" << cmd;
99 if (writeType)
100 {
101 t << " type=\"";
102 switch(type)
103 {
104 case DocImage::Html: t << "html"; break;
105 case DocImage::Latex: t << "latex"; break;
106 case DocImage::Rtf: t << "rtf"; break;
107 case DocImage::DocBook: t << "docbook"; break;
108 case DocImage::Xml: t << "xml"; break;
109 }
110 t << "\"";
111 }
112 if (!name.isEmpty())
113 {
114 t << " name=\"" << convertToXML(name, TRUE) << "\"";
115 }
116 if (!width.isEmpty())
117 {
118 t << " width=\"" << convertToXML(width) << "\"";
119 }
120 if (!height.isEmpty())
121 {
122 t << " height=\"" << convertToXML(height) << "\"";
123 }
124 if (!engine.isEmpty())
125 {
126 t << " engine=\"" << convertToXML(engine) << "\"";
127 }
128 if (!alt.isEmpty())
129 {
130 t << " alt=\"" << convertToXML(alt) << "\"";
131 }
132 if (inlineImage)
133 {
134 t << " inline=\"yes\"";
135 }
136 if (doCaption)
137 {
138 t << " caption=\"";
139 visitCaption(visitor, children);
140 t << "\"";
141 }
142 t << ">";
143}

145static void visitPostEnd(TextStream &t, const char *cmd)
146{
147 t << "</" << cmd << ">\n";
148}

303void XmlDocVisitor::operator()(const DocVerbatim &s)
304{
305 if (m_hide) return;
306 QCString lang = m_langExt;
307 if (!s.language().isEmpty()) // explicit language setting
308 {
309 lang = s.language();
310 }
311 SrcLangExt langExt = getLanguageFromCodeLang(lang);
312 switch(s.type())
313 {
314 case DocVerbatim::Code:
315 m_t << "<programlisting";
316 if (!s.language().isEmpty())
317 m_t << " filename=\"" << lang << "\">";
318 else
319 m_t << ">";
320 getCodeParser(lang).parseCode(m_ci,s.context(),s.text(),langExt,
321 Config_getBool(STRIP_CODE_COMMENTS),
322 s.isExample(),s.exampleFile());
323 m_t << "</programlisting>";
324 break;
325 case DocVerbatim::JavaDocLiteral:
326 m_t << "<javadocliteral>";
327 filter(s.text());
328 m_t << "</javadocliteral>";
329 break;
330 case DocVerbatim::JavaDocCode:
331 m_t << "<javadoccode>";
332 filter(s.text());
333 m_t << "</javadoccode>";
334 break;
335 case DocVerbatim::Verbatim:
336 m_t << "<verbatim>";
337 filter(s.text());
338 m_t << "</verbatim>";
339 break;
340 case DocVerbatim::HtmlOnly:
341 if (s.isBlock())
342 {
343 m_t << "<htmlonly block=\"yes\">";
344 }
345 else
346 {
347 m_t << "<htmlonly>";
348 }
349 filter(s.text());
350 m_t << "</htmlonly>";
351 break;
352 case DocVerbatim::RtfOnly:
353 m_t << "<rtfonly>";
354 filter(s.text());
355 m_t << "</rtfonly>";
356 break;
357 case DocVerbatim::ManOnly:
358 m_t << "<manonly>";
359 filter(s.text());
360 m_t << "</manonly>";
361 break;
362 case DocVerbatim::LatexOnly:
363 m_t << "<latexonly>";
364 filter(s.text());
365 m_t << "</latexonly>";
366 break;
367 case DocVerbatim::DocbookOnly:
368 m_t << "<docbookonly>";
369 filter(s.text());
370 m_t << "</docbookonly>";
371 break;
372 case DocVerbatim::XmlOnly:
373 m_t << s.text();
374 break;
375 case DocVerbatim::Dot:
376 visitPreStart(m_t, "dot", s.hasCaption(), *this, s.children(), QCString(""), FALSE, DocImage::Html, s.width(), s.height());
377 filter(s.text());
378 visitPostEnd(m_t, "dot");
379 break;
380 case DocVerbatim::Msc:
381 visitPreStart(m_t, "msc", s.hasCaption(), *this, s.children(), QCString(""), FALSE, DocImage::Html, s.width(), s.height());
382 filter(s.text());
383 visitPostEnd(m_t, "msc");
384 break;
385 case DocVerbatim::PlantUML:
386 visitPreStart(m_t, "plantuml", s.hasCaption(), *this, s.children(), QCString(""), FALSE, DocImage::Html, s.width(), s.height(), s.engine());
387 filter(s.text());
388 visitPostEnd(m_t, "plantuml");
389 break;
390 }
391}

399void XmlDocVisitor::operator()(const DocInclude &inc)
400{
401 if (m_hide) return;
402 SrcLangExt langExt = getLanguageFromFileName(inc.extension());
403 //printf("XMLDocVisitor: DocInclude type=%d trimleft=%d\n",inc.type(),inc.trimLeft());
404 switch(inc.type())
405 {
406 case DocInclude::IncWithLines:
407 {
408 m_t << "<programlisting filename=\"" << inc.file() << "\">";
409 FileInfo cfi( inc.file().str() );
410 auto fd = createFileDef( cfi.dirPath(), cfi.fileName());
411 getCodeParser(inc.extension()).parseCode(m_ci,inc.context(),
412 inc.text(),
413 langExt,
414 inc.stripCodeComments(),
415 inc.isExample(),
416 inc.exampleFile(),
417 fd.get(), // fileDef,
418 -1, // start line
419 -1, // end line
420 FALSE, // inline fragment
421 nullptr, // memberDef
422 TRUE // show line numbers
423 );
424 m_t << "</programlisting>";
425 }
426 break;
427 case DocInclude::Include:
428 m_t << "<programlisting filename=\"" << inc.file() << "\">";
429 getCodeParser(inc.extension()).parseCode(m_ci,inc.context(),
430 inc.text(),
431 langExt,
432 inc.stripCodeComments(),
433 inc.isExample(),
434 inc.exampleFile(),
435 nullptr, // fileDef
436 -1, // startLine
437 -1, // endLine
438 TRUE, // inlineFragment
439 nullptr, // memberDef
440 FALSE // show line numbers
441 );
442 m_t << "</programlisting>";
443 break;
444 case DocInclude::DontInclude:
445 case DocInclude::DontIncWithLines:
446 break;
447 case DocInclude::HtmlInclude:
448 if (inc.isBlock())
449 {
450 m_t << "<htmlonly block=\"yes\">";
451 }
452 else
453 {
454 m_t << "<htmlonly>";
455 }
456 filter(inc.text());
457 m_t << "</htmlonly>";
458 break;
459 case DocInclude::LatexInclude:
460 m_t << "<latexonly>";
461 filter(inc.text());
462 m_t << "</latexonly>";
463 break;
464 case DocInclude::RtfInclude:
465 m_t << "<rtfonly>";
466 filter(inc.text());
467 m_t << "</rtfonly>";
468 break;
469 case DocInclude::ManInclude:
470 m_t << "<manonly>";
471 filter(inc.text());
472 m_t << "</manonly>";
473 break;
474 case DocInclude::XmlInclude:
475 filter(inc.text());
476 break;
477 case DocInclude::DocbookInclude:
478 m_t << "<docbookonly>";
479 filter(inc.text());
480 m_t << "</docbookonly>";
481 break;
482 case DocInclude::VerbInclude:
483 m_t << "<verbatim>";
484 filter(inc.text());
485 m_t << "</verbatim>";
486 break;
487 case DocInclude::Snippet:
488 case DocInclude::SnippetWithLines:
489 m_t << "<programlisting filename=\"" << inc.file() << "\">";
490 CodeFragmentManager::instance().parseCodeFragment(m_ci,
491 inc.file(),
492 inc.blockId(),
493 inc.context(),
494 inc.type()==DocInclude::SnippetWithLines,
495 inc.trimLeft(),
496 inc.stripCodeComments()
497 );
498 m_t << "</programlisting>";
499 break;
500 }
501}

503void XmlDocVisitor::operator()(const DocIncOperator &op)
504{
505 //printf("DocIncOperator: type=%d first=%d, last=%d text='%s'\n",
506 // op.type(),op.isFirst(),op.isLast(),qPrint(op.text()));
507 if (op.isFirst())
508 {
509 if (!m_hide)
510 {
511 m_t << "<programlisting filename=\"" << op.includeFileName() << "\">";
512 }
513 pushHidden(m_hide);
514 m_hide = TRUE;
515 }
516 QCString locLangExt = getFileNameExtension(op.includeFileName());
517 if (locLangExt.isEmpty()) locLangExt = m_langExt;
518 SrcLangExt langExt = getLanguageFromFileName(locLangExt);
519 if (op.type()!=DocIncOperator::Skip)
520 {
521 m_hide = popHidden();
522 if (!m_hide)
523 {
524 std::unique_ptr<FileDef> fd;
525 if (!op.includeFileName().isEmpty())
526 {
527 FileInfo cfi( op.includeFileName().str() );
528 fd = createFileDef( cfi.dirPath(), cfi.fileName() );
529 }
530
531 getCodeParser(locLangExt).parseCode(m_ci,op.context(),
532 op.text(),langExt,
533 op.stripCodeComments(),
534 op.isExample(),
535 op.exampleFile(),
536 fd.get(), // fileDef
537 op.line(), // startLine
538 -1, // endLine
539 FALSE, // inline fragment
540 nullptr, // memberDef
541 op.showLineNo() // show line numbers
542 );
543 }
544 pushHidden(m_hide);
545 m_hide=TRUE;
546 }
547 if (op.isLast())
548 {
549 m_hide = popHidden();
550 if (!m_hide) m_t << "</programlisting>";
551 }
552 else
553 {
554 if (!m_hide) m_t << "\n";
555 }
556}

693void XmlDocVisitor::operator()(const DocSection &s)
694{
695 if (m_hide) return;
696 int orgSectionLevel = m_sectionLevel;
697 QCString sectId = s.file();
698 if (!s.anchor().isEmpty()) sectId += "_1"+s.anchor();
699 while (m_sectionLevel+1<s.level()) // fix missing intermediate levels
700 {
701 m_sectionLevel++;
702 m_t << "<sect" << m_sectionLevel << " id=\"" << sectId << "_1s" << m_sectionLevel << "\">";
703 }
704 m_sectionLevel++;
705 m_t << "<sect" << s.level() << " id=\"" << sectId << "\">\n";
706 if (s.title())
707 {
708 std::visit(*this,*s.title());
709 }
710 visitChildren(s);
711 m_t << "</sect" << s.level() << ">";
712 m_sectionLevel--;
713 while (orgSectionLevel<m_sectionLevel) // fix missing intermediate levels
714 {
715 m_t << "</sect" << m_sectionLevel << ">";
716 m_sectionLevel--;
717 }
718 m_t << "\n";
719}
