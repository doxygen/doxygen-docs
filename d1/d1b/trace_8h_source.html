55 inline QCString trunc(const QCString &s,size_t numChars=15)
56 {
57 QCString result;
58 size_t i=0;
59 for (; i<numChars && i<s.length(); i++)
60 {
61 char c=s.at(i);
62 if (c=='\n') { result+="\\n"; }
63 else if (c=='\t') { result+="\\t"; }
64 else if (c=='\r') { result+="\\r"; }
65 else if (c=='\\') { result+="\\\\"; }
66 else result+=c;
67 }
68 if (i<s.length()) result+="...";
69 return result;
70 }
71}

79/** Helper class to trace an entry statement at creation and another one at destruction. */
80class AutoTrace
81{
82 public:
83 explicit AutoTrace(spdlog::source_loc loc) : m_loc(loc)
84 {
85 if (g_tracer)
86 {
87 g_tracer->log(m_loc,spdlog::level::trace,">");
88 }
89 }
90 template<typename... Args>
91 explicit AutoTrace(spdlog::source_loc loc,
92 const std::string &fmt, Args&&...args) : m_loc(loc)
93 {
94 if (g_tracer)
95 {
96 if (fmt.empty())
97 {
98 g_tracer->log(m_loc,spdlog::level::trace,">");
99 }
100 else
101 {
102 g_tracer->log(m_loc,spdlog::level::trace,"> "+fmt,std::forward<Args>(args)...);
103 }
104 }
105 }
106 ~AutoTrace()
107 {
108 if (g_tracer)
109 {
110 if (m_exitMessage.empty())
111 {
112 g_tracer->log(m_loc,spdlog::level::trace,"<");
113 }
114 else
115 {
116 g_tracer->log(m_loc,spdlog::level::trace,"< "+m_exitMessage);
117 }
118 }
119 }
120 template<typename... Args>
121 void add(spdlog::source_loc loc,
122 const std::string &fmt, Args&&...args)
123 {
124 if (g_tracer)
125 {
126 g_tracer->log(loc,spdlog::level::trace,": "+fmt,std::forward<Args>(args)...);
127 }
128 }
129 template<typename... Args>
130 void setExit(const std::string &msg,Args&&...args)
131 {
132 m_exitMessage = fmt::format(msg,std::forward<Args>(args)...);
133 }
134 private:
135 spdlog::source_loc m_loc;
136 std::string m_exitMessage;
137};
