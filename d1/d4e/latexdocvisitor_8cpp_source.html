58const char *LatexDocVisitor::getSectionName(int level) const
59{
60 bool compactLatex = Config_getBool(COMPACT_LATEX);
61 int l = level;
62 if (compactLatex) l++;
63
64 if (l < g_maxLevels)
65 {
66 l += m_hierarchyLevel; /* May be -1 if generating main page */
67 // Sections get special treatment because they inherit the parent's level
68 if (l >= g_maxLevels)
69 {
70 l = g_maxLevels - 1;
71 }
72 else if (l < 0)
73 {
74 /* Should not happen; level is always >= 1 and hierarchyLevel >= -1 */
75 l = 0;
76 }
77 return g_secLabels[l];
78 }
79 else if (l == 7)
80 {
81 return g_paragraphLabel;
82 }
83 else
84 {
85 return g_subparagraphLabel;
86 }
87}

89static void insertDimension(TextStream &t, QCString dimension, const char *orientationString)
90{
91 // dimensions for latex images can be a percentage, in this case they need some extra
92 // handling as the % symbol is used for comments
93 static const reg::Ex re(R"((\d+)%)");
94 std::string s = dimension.str();
95 reg::Match match;
96 if (reg::search(s,match,re))
97 {
98 bool ok = false;
99 double percent = QCString(match[1].str()).toInt(&ok);
100 if (ok)
101 {
102 t << percent/100.0 << "\\text" << orientationString;
103 return;
104 }
105 }
106 t << dimension;
107}

109static void visitPreStart(TextStream &t, bool hasCaption, QCString name, QCString width, QCString height, bool inlineImage = FALSE)
110{
111 if (inlineImage)
112 {
113 t << "\n\\begin{DoxyInlineImage}\n";
114 }
115 else
116 {
117 if (hasCaption)
118 {
119 t << "\n\\begin{DoxyImage}\n";
120 }
121 else
122 {
123 t << "\n\\begin{DoxyImageNoCaption}\n"
124 " \\mbox{";
125 }
126 }
127
128 t << "\\includegraphics";
129 if (!width.isEmpty() || !height.isEmpty())
130 {
131 t << "[";
132 }
133 if (!width.isEmpty())
134 {
135 t << "width=";
136 insertDimension(t, width, "width");
137 }
138 if (!width.isEmpty() && !height.isEmpty())
139 {
140 t << ",";
141 }
142 if (!height.isEmpty())
143 {
144 t << "height=";
145 insertDimension(t, height, "height");
146 }
147 if (width.isEmpty() && height.isEmpty())
148 {
149 /* default setting */
150 if (inlineImage)
151 {
152 t << "[height=\\baselineskip,keepaspectratio=true]";
153 }
154 else
155 {
156 t << "[width=\\textwidth,height=\\textheight/2,keepaspectratio=true]";
157 }
158 }
159 else
160 {
161 t << "]";
162 }
163
164 t << "{" << name << "}";
165
166 if (hasCaption)
167 {
168 if (!inlineImage)
169 {
170 if (Config_getBool(PDF_HYPERLINKS))
171 {
172 t << "\n\\doxyfigcaption{";
173 }
174 else
175 {
176 t << "\n\\doxyfigcaptionnolink{";
177 }
178 }
179 else
180 {
181 t << "%"; // to catch the caption
182 }
183 }
184}

239QCString LatexDocVisitor::escapeMakeIndexChars(const char *s)
240{
241 QCString result;
242 const char *p=s;
243 char str[2]; str[1]=0;
244 char c = 0;
245 if (p)
246 {
247 while ((c=*p++))
248 {
249 switch (c)
250 {
251 case '!': m_t << "\"!"; break;
252 case '"': m_t << "\"\""; break;
253 case '@': m_t << "\"@"; break;
254 case '|': m_t << "\\texttt{\"|}"; break;
255 case '[': m_t << "["; break;
256 case ']': m_t << "]"; break;
257 case '{': m_t << "\\lcurly{}"; break;
258 case '}': m_t << "\\rcurly{}"; break;
259 default: str[0]=c; filter(str); break;
260 }
261 }
262 }
263 return result;
264}

447void LatexDocVisitor::operator()(const DocVerbatim &s)
448{
449 if (m_hide) return;
450 QCString lang = m_langExt;
451 if (!s.language().isEmpty()) // explicit language setting
452 {
453 lang = s.language();
454 }
455 SrcLangExt langExt = getLanguageFromCodeLang(lang);
456 switch(s.type())
457 {
458 case DocVerbatim::Code:
459 {
460 m_ci.startCodeFragment("DoxyCode");
461 getCodeParser(lang).parseCode(m_ci,s.context(),s.text(),langExt,
462 Config_getBool(STRIP_CODE_COMMENTS),
463 s.isExample(),s.exampleFile());
464 m_ci.endCodeFragment("DoxyCode");
465 }
466 break;
467 case DocVerbatim::JavaDocLiteral:
468 filter(s.text(), true);
469 break;
470 case DocVerbatim::JavaDocCode:
471 m_t << "{\\ttfamily ";
472 filter(s.text(), true);
473 m_t << "}";
474 break;
475 case DocVerbatim::Verbatim:
476 m_t << "\\begin{DoxyVerb}";
477 m_t << s.text();
478 m_t << "\\end{DoxyVerb}\n";
479 break;
480 case DocVerbatim::HtmlOnly:
481 case DocVerbatim::XmlOnly:
482 case DocVerbatim::ManOnly:
483 case DocVerbatim::RtfOnly:
484 case DocVerbatim::DocbookOnly:
485 /* nothing */
486 break;
487 case DocVerbatim::LatexOnly:
488 m_t << s.text();
489 break;
490 case DocVerbatim::Dot:
491 {
492 static int dotindex = 1;
493 QCString fileName(4096, QCString::ExplicitSize);
494
495 fileName.sprintf("%s%d%s",
496 qPrint(Config_getString(LATEX_OUTPUT)+"/inline_dotgraph_"),
497 dotindex++,
498 ".dot"
499 );
500 std::ofstream file = Portable::openOutputStream(fileName);
501 if (!file.is_open())
502 {
503 err("Could not open file %s for writing\n",qPrint(fileName));
504 }
505 else
506 {
507 file.write( s.text().data(), s.text().length() );
508 file.close();
509
510 startDotFile(fileName,s.width(),s.height(),s.hasCaption(),s.srcFile(),s.srcLine());
511 visitChildren(s);
512 endDotFile(s.hasCaption());
513
514 if (Config_getBool(DOT_CLEANUP)) Dir().remove(fileName.str());
515 }
516 }
517 break;
518 case DocVerbatim::Msc:
519 {
520 static int mscindex = 1;
521 QCString baseName(4096, QCString::ExplicitSize);
522
523 baseName.sprintf("%s%d",
524 qPrint(Config_getString(LATEX_OUTPUT)+"/inline_mscgraph_"),
525 mscindex++
526 );
527 QCString fileName = baseName+".msc";
528 std::ofstream file = Portable::openOutputStream(fileName);
529 if (!file.is_open())
530 {
531 err("Could not open file %s for writing\n",qPrint(fileName));
532 }
533 else
534 {
535 QCString text = "msc {";
536 text+=s.text();
537 text+="}";
538 file.write( text.data(), text.length() );
539 file.close();
540
541 writeMscFile(baseName, s);
542
543 if (Config_getBool(DOT_CLEANUP)) Dir().remove(fileName.str());
544 }
545 }
546 break;
547 case DocVerbatim::PlantUML:
548 {
549 QCString latexOutput = Config_getString(LATEX_OUTPUT);
550 QCString baseName = PlantumlManager::instance().writePlantUMLSource(
551 latexOutput,s.exampleFile(),s.text(),
552 s.useBitmap() ? PlantumlManager::PUML_BITMAP : PlantumlManager::PUML_EPS,
553 s.engine(),s.srcFile(),s.srcLine(),true);
554
555 writePlantUMLFile(baseName, s);
556 }
557 break;
558 }
559}

572void LatexDocVisitor::operator()(const DocInclude &inc)
573{
574 if (m_hide) return;
575 SrcLangExt langExt = getLanguageFromFileName(inc.extension());
576 switch(inc.type())
577 {
578 case DocInclude::IncWithLines:
579 {
580 m_ci.startCodeFragment("DoxyCodeInclude");
581 FileInfo cfi( inc.file().str() );
582 auto fd = createFileDef( cfi.dirPath(), cfi.fileName() );
583 getCodeParser(inc.extension()).parseCode(m_ci,inc.context(),
584 inc.text(),
585 langExt,
586 inc.stripCodeComments(),
587 inc.isExample(),
588 inc.exampleFile(),
589 fd.get(), // fileDef,
590 -1, // start line
591 -1, // end line
592 FALSE, // inline fragment
593 nullptr, // memberDef
594 TRUE // show line numbers
595 );
596 m_ci.endCodeFragment("DoxyCodeInclude");
597 }
598 break;
599 case DocInclude::Include:
600 {
601 m_ci.startCodeFragment("DoxyCodeInclude");
602 getCodeParser(inc.extension()).parseCode(m_ci,inc.context(),
603 inc.text(),langExt,
604 inc.stripCodeComments(),
605 inc.isExample(),
606 inc.exampleFile(),
607 nullptr, // fileDef
608 -1, // startLine
609 -1, // endLine
610 TRUE, // inlineFragment
611 nullptr, // memberDef
612 FALSE
613 );
614 m_ci.endCodeFragment("DoxyCodeInclude");
615 }
616 break;
617 case DocInclude::DontInclude:
618 case DocInclude::DontIncWithLines:
619 case DocInclude::HtmlInclude:
620 case DocInclude::RtfInclude:
621 case DocInclude::ManInclude:
622 case DocInclude::XmlInclude:
623 case DocInclude::DocbookInclude:

634 case DocInclude::SnippetWithLines:
635 {
636 m_ci.startCodeFragment("DoxyCodeInclude");
637 CodeFragmentManager::instance().parseCodeFragment(m_ci,
638 inc.file(),
639 inc.blockId(),
640 inc.context(),
641 inc.type()==DocInclude::SnippetWithLines,
642 inc.trimLeft(),
643 inc.stripCodeComments()
644 );
645 m_ci.endCodeFragment("DoxyCodeInclude");
646 }
647 break;
648 }

651void LatexDocVisitor::operator()(const DocIncOperator &op)
652{
653 //printf("DocIncOperator: type=%d first=%d, last=%d text='%s'\n",
654 // op.type(),op.isFirst(),op.isLast(),qPrint(op.text()));
655 if (op.isFirst())
656 {
657 if (!m_hide) m_ci.startCodeFragment("DoxyCodeInclude");
658 pushHidden(m_hide);
659 m_hide = TRUE;
660 }
661 QCString locLangExt = getFileNameExtension(op.includeFileName());
662 if (locLangExt.isEmpty()) locLangExt = m_langExt;
663 SrcLangExt langExt = getLanguageFromFileName(locLangExt);
664 if (op.type()!=DocIncOperator::Skip)
665 {
666 m_hide = popHidden();
667 if (!m_hide)
668 {
669 std::unique_ptr<FileDef> fd;
670 if (!op.includeFileName().isEmpty())
671 {
672 FileInfo cfi( op.includeFileName().str() );
673 fd = createFileDef( cfi.dirPath(), cfi.fileName() );
674 }
675
676 getCodeParser(locLangExt).parseCode(m_ci,op.context(),op.text(),langExt,
677 op.stripCodeComments(),
678 op.isExample(),op.exampleFile(),
679 fd.get(), // fileDef
680 op.line(), // startLine
681 -1, // endLine
682 FALSE, // inline fragment
683 nullptr, // memberDef
684 op.showLineNo() // show line numbers
685 );
686 }
687 pushHidden(m_hide);
688 m_hide=TRUE;
689 }
690 if (op.isLast())
691 {
692 m_hide=popHidden();
693 if (!m_hide) m_ci.endCodeFragment("DoxyCodeInclude");
694 }
695 else
696 {
697 if (!m_hide) m_t << "\n";
698 }
699}

1280{
1281 if (m_hide) return;
1282 setCurrentColumn(0);
1283
1284 visitChildren(row);
1285
1286 size_t c=currentColumn();

1288 {

1291 //printf(" found row span: column=%d rs=%d cs=%d rowIdx=%d cell->rowIdx=%d i=%d c=%d\n",
1292 // span->column, span->rowSpan,span->colSpan,row.rowIndex(),span->cell->rowIndex(),i,c);
1293 if (span.rowSpan>0 && span.column==c && // we are at a cell in a row span
1294 row.rowIndex()>span.cell.rowIndex() // but not the row that started the span
1295 )

1302 }
1303 else // solitary row span
1304 {
1305 m_t << "\\multicolumn{1}{c|}{}";
1306 }
1307 }
1308 }

1310 }
1311

1315 for (auto &span : rowSpans())
1316 {
1317 if (span.rowSpan>0) span.rowSpan--;
1318 if (span.rowSpan<=0)
1319 {
1320 // inactive span
1321 }

1323 {

1325 col = span.column+span.colSpan;
