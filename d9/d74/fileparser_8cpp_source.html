42 size_t j=i;

45 codeOutIntf.startCodeLine(lineNr);

48 codeOutIntf.writeLineNumber(QCString(),QCString(),QCString(),lineNr,!inlineFragment);
49 }

51 codeOutIntf.endCodeLine();
52 lineNr++;
53 i=j+1;
54 }
