342//! Return the index of the last :: in the string \a name that is still before the first <
343inline int computeQualifiedIndex(const QCString &name)
344{
345 int l = static_cast<int>(name.length());
346 int lastSepPos = -1;
347 const char *p = name.data();
348 for (int i=0;i<l-1;i++)
349 {
350 char c=*p++;
351 if (c==':' && *p==':') lastSepPos=i;
352 if (c=='<') break;
353 }
354 return lastSepPos;
355}
