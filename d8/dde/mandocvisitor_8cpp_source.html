127void ManDocVisitor::operator()(const DocStyleChange &s)
128{
129 if (m_hide) return;
130 switch (s.style())
131 {
132 case DocStyleChange::Bold:
133 if (s.enable()) m_t << "\\fB"; else m_t << "\\fP";
134 m_firstCol=FALSE;
135 break;
136 case DocStyleChange::S:
137 case DocStyleChange::Strike:
138 case DocStyleChange::Del:
139 /* not supported */
140 break;
141 case DocStyleChange::Underline: //underline is shown as emphasis
142 case DocStyleChange::Ins:
143 if (s.enable()) m_t << "\\fI"; else m_t << "\\fP";
144 m_firstCol=FALSE;
145 break;
146 case DocStyleChange::Italic:
147 if (s.enable()) m_t << "\\fI"; else m_t << "\\fP";
148 m_firstCol=FALSE;
149 break;
150 case DocStyleChange::Code:
151 if (s.enable()) m_t << "\\fR"; else m_t << "\\fP";
152 m_firstCol=FALSE;
153 break;
154 case DocStyleChange::Subscript:
155 if (s.enable()) m_t << "\\*<"; else m_t << "\\*> ";
156 m_firstCol=FALSE;
157 break;
158 case DocStyleChange::Superscript:
159 if (s.enable()) m_t << "\\*{"; else m_t << "\\*} ";
160 m_firstCol=FALSE;
161 break;
162 case DocStyleChange::Center:
163 /* not supported */
164 break;
165 case DocStyleChange::Small:
166 /* not supported */
167 break;
168 case DocStyleChange::Cite:
169 /* not supported */
170 break;
171 case DocStyleChange::Preformatted:
172 if (s.enable())
173 {
174 if (!m_firstCol) m_t << "\n";
175 m_t << ".PP\n";
176 m_t << ".nf\n";
177 m_insidePre=TRUE;
178 }
179 else
180 {
181 m_insidePre=FALSE;
182 if (!m_firstCol) m_t << "\n";
183 m_t << ".fi\n";
184 m_t << ".PP\n";
185 m_firstCol=TRUE;
186 }
187 break;
188 case DocStyleChange::Div: /* HTML only */ break;
189 case DocStyleChange::Span: /* HTML only */ break;
190 }
191}

193void ManDocVisitor::operator()(const DocVerbatim &s)
194{
195 if (m_hide) return;
196 QCString lang = m_langExt;
197 if (!s.language().isEmpty()) // explicit language setting
198 {
199 lang = s.language();
200 }
201 SrcLangExt langExt = getLanguageFromCodeLang(lang);
202 switch (s.type())
203 {
204 case DocVerbatim::Code:
205 if (!m_firstCol) m_t << "\n";
206 m_t << ".PP\n";
207 m_t << ".nf\n";
208 getCodeParser(lang).parseCode(m_ci,s.context(),s.text(),
209 langExt,
210 Config_getBool(STRIP_CODE_COMMENTS),
211 s.isExample(),s.exampleFile());
212 if (!m_firstCol) m_t << "\n";
213 m_t << ".fi\n";
214 m_t << ".PP\n";
215 m_firstCol=TRUE;
216 break;
217 case DocVerbatim::JavaDocLiteral:
218 filter(s.text());
219 break;
220 case DocVerbatim::JavaDocCode:
221 m_t << "\\fR\n";
222 filter(s.text());
223 m_t << "\\fP\n";
224 break;
225 case DocVerbatim::Verbatim:
226 if (!m_firstCol) m_t << "\n";
227 m_t << ".PP\n";
228 m_t << ".nf\n";
229 filter(s.text());
230 if (!m_firstCol) m_t << "\n";
231 m_t << ".fi\n";
232 m_t << ".PP\n";
233 m_firstCol=TRUE;
234 break;
235 case DocVerbatim::ManOnly:
236 m_t << s.text();
237 break;
238 case DocVerbatim::HtmlOnly:
239 case DocVerbatim::XmlOnly:
240 case DocVerbatim::LatexOnly:
241 case DocVerbatim::RtfOnly:
242 case DocVerbatim::DocbookOnly:
243 case DocVerbatim::Dot:
244 case DocVerbatim::Msc:
245 case DocVerbatim::PlantUML:
246 /* nothing */
247 break;
248 }

256void ManDocVisitor::operator()(const DocInclude &inc)
257{
258 if (m_hide) return;
259 SrcLangExt langExt = getLanguageFromFileName(inc.extension());
260 switch(inc.type())
261 {
262 case DocInclude::IncWithLines:
263 {
264 if (!m_firstCol) m_t << "\n";
265 m_t << ".PP\n";
266 m_t << ".nf\n";
267 FileInfo cfi( inc.file().str() );
268 auto fd = createFileDef( cfi.dirPath(), cfi.fileName() );
269 getCodeParser(inc.extension()).parseCode(m_ci,inc.context(),
270 inc.text(),
271 langExt,
272 inc.stripCodeComments(),
273 inc.isExample(),
274 inc.exampleFile(),
275 fd.get(), // fileDef,
276 -1, // start line
277 -1, // end line
278 FALSE, // inline fragment
279 nullptr, // memberDef
280 TRUE
281 );
282 m_t << ".fi\n";
283 m_t << ".PP\n";
284 m_firstCol=TRUE;
285 }
286 break;
287 case DocInclude::Include:
288 if (!m_firstCol) m_t << "\n";
289 m_t << ".PP\n";
290 m_t << ".nf\n";
291 getCodeParser(inc.extension()).parseCode(m_ci,inc.context(),
292 inc.text(),
293 langExt,
294 inc.stripCodeComments(),
295 inc.isExample(),
296 inc.exampleFile(),
297 nullptr, // fileDef
298 -1, // startLine
299 -1, // endLine
300 TRUE, // inlineFragment
301 nullptr, // memberDef
302 FALSE
303 );
304 m_t << ".fi\n";
305 m_t << ".PP\n";
306 m_firstCol=TRUE;
307 break;
308 case DocInclude::DontInclude:
309 case DocInclude::DontIncWithLines:
310 case DocInclude::HtmlInclude:
311 case DocInclude::LatexInclude:
312 case DocInclude::RtfInclude:
313 case DocInclude::XmlInclude:
314 case DocInclude::DocbookInclude:
315 break;
316 case DocInclude::ManInclude:
317 m_t << inc.text();
318 break;
319 case DocInclude::VerbInclude:
320 if (!m_firstCol) m_t << "\n";
321 m_t << ".PP\n";
322 m_t << ".nf\n";
323 m_t << inc.text();
324 if (!m_firstCol) m_t << "\n";
325 m_t << ".fi\n";
326 m_t << ".PP\n";
327 m_firstCol=TRUE;
328 break;
329 case DocInclude::Snippet:
330 case DocInclude::SnippetWithLines:
331 if (!m_firstCol) m_t << "\n";
332 m_t << ".PP\n";
333 m_t << ".nf\n";
334 CodeFragmentManager::instance().parseCodeFragment(m_ci,
335 inc.file(),
336 inc.blockId(),
337 inc.context(),
338 inc.type()==DocInclude::SnippetWithLines,
339 inc.trimLeft(),
340 inc.stripCodeComments()
341 );
342 m_t << ".fi\n";
343 m_t << ".PP\n";
344 m_firstCol=TRUE;
345 break;
346 }
347}

349void ManDocVisitor::operator()(const DocIncOperator &op)
350{
351 QCString locLangExt = getFileNameExtension(op.includeFileName());
352 if (locLangExt.isEmpty()) locLangExt = m_langExt;
353 SrcLangExt langExt = getLanguageFromFileName(locLangExt);
354 //printf("DocIncOperator: type=%d first=%d, last=%d text='%s'\n",
355 // op.type(),op.isFirst(),op.isLast(),qPrint(op.text()));
356 if (op.isFirst())
357 {
358 if (!m_hide)
359 {
360 if (!m_firstCol) m_t << "\n";
361 m_t << ".PP\n";
362 m_t << ".nf\n";
363 }
364 pushHidden(m_hide);
365 m_hide = TRUE;
366 }
367 if (op.type()!=DocIncOperator::Skip)
368 {
369 m_hide = popHidden();
370 if (!m_hide)
371 {
372 std::unique_ptr<FileDef> fd;
373 if (!op.includeFileName().isEmpty())
374 {
375 FileInfo cfi( op.includeFileName().str() );
376 fd = createFileDef( cfi.dirPath(), cfi.fileName() );
377 }
378
379 getCodeParser(locLangExt).parseCode(m_ci,op.context(),op.text(),langExt,
380 op.stripCodeComments(),
381 op.isExample(),op.exampleFile(),
382 fd.get(), // fileDef
383 op.line(), // startLine
384 -1, // endLine
385 FALSE, // inline fragment
386 nullptr, // memberDef
387 op.showLineNo() // show line numbers
388 );
389 }
390 pushHidden(m_hide);
391 m_hide=TRUE;
392 }
393 if (op.isLast())
394 {
395 m_hide = popHidden();
396 if (!m_hide)
397 {
398 if (!m_firstCol) m_t << "\n";
399 m_t << ".fi\n";
400 m_t << ".PP\n";
401 m_firstCol=TRUE;
402 }
403 }
404 else
405 {
406 if (!m_hide) m_t << "\n";
407 }
408}

454void ManDocVisitor::operator()(const DocAutoListItem &li)
455{
456 if (m_hide) return;
457 QCString ws;
458 if (m_indent>0) ws.fill(' ',2*(m_indent-1));
459 if (!m_firstCol) m_t << "\n";
460 m_t << ".IP \"" << ws;
461 const DocAutoList *list = std::get_if<DocAutoList>(li.parent());
462 if (list && list->isEnumList())
463 {
464 m_t << li.itemNumber() << ".\" " << (2*(m_indent+1));
465 }
466 else // bullet list
467 {
468 switch (li.itemNumber())
469 {
470 case DocAutoList::Unchecked: // unchecked
471 m_t << "[ ]\" " << (2*m_indent) + 2;
472 break;
473 case DocAutoList::Checked_x: // checked with x
474 m_t << "[x]\" " << (2*m_indent) + 2;
475 break;
476 case DocAutoList::Checked_X: // checked with X
477 m_t << "[X]\" " << (2*m_indent) + 2;
478 break;
479 default:
480 m_t << "\\(bu\" " << (2*m_indent);
481 break;
482 }
483 }
484 m_t << "\n";
485 m_firstCol=TRUE;
486 visitChildren(li);
487 m_t << "\n";
488 m_firstCol=TRUE;
489}

512void ManDocVisitor::operator()(const DocSimpleSect &s)
513{
514 if (m_hide) return;
515 if (!m_firstCol)
516 {
517 m_t << "\n";
518 m_t << ".PP\n";
519 }
520 m_t << "\\fB";
521 switch(s.type())
522 {
523 case DocSimpleSect::See:
524 m_t << theTranslator->trSeeAlso(); break;
525 case DocSimpleSect::Return:
526 m_t << theTranslator->trReturns(); break;
527 case DocSimpleSect::Author:
528 m_t << theTranslator->trAuthor(TRUE,TRUE); break;
529 case DocSimpleSect::Authors:
530 m_t << theTranslator->trAuthor(TRUE,FALSE); break;
531 case DocSimpleSect::Version:
532 m_t << theTranslator->trVersion(); break;
533 case DocSimpleSect::Since:
534 m_t << theTranslator->trSince(); break;
535 case DocSimpleSect::Date:
536 m_t << theTranslator->trDate(); break;
537 case DocSimpleSect::Note:
538 m_t << theTranslator->trNote(); break;
539 case DocSimpleSect::Warning:
540 m_t << theTranslator->trWarning(); break;
541 case DocSimpleSect::Pre:
542 m_t << theTranslator->trPrecondition(); break;
543 case DocSimpleSect::Post:
544 m_t << theTranslator->trPostcondition(); break;
545 case DocSimpleSect::Copyright:
546 m_t << theTranslator->trCopyright(); break;
547 case DocSimpleSect::Invar:
548 m_t << theTranslator->trInvariant(); break;
549 case DocSimpleSect::Remark:
550 m_t << theTranslator->trRemarks(); break;
551 case DocSimpleSect::Attention:
552 m_t << theTranslator->trAttention(); break;
553 case DocSimpleSect::Important:

555 case DocSimpleSect::User: break;
556 case DocSimpleSect::Rcs: break;
557 case DocSimpleSect::Unknown: break;

560 // special case 1: user defined title
561 if (s.title())
562 {
563 std::visit(*this,*s.title());

565 m_t << "\\fP\n";
566 m_t << ".RS 4\n";
567 visitChildren(s);
568 if (!m_firstCol) m_t << "\n";
569 m_t << ".RE\n";
570 m_t << ".PP\n";
571 m_firstCol=TRUE;
572}

624void ManDocVisitor::operator()(const DocHtmlList &l)
625{
626 if (m_hide) return;
627 m_indent++;
628 if (!m_firstCol) m_t << "\n";
629 m_t << ".PD 0\n";
630 m_firstCol=true;
631 int indent = std::min(m_indent,maxIndentLevels-1);
632 m_listItemInfo[indent].number = 1;
633 m_listItemInfo[indent].type = '1';
634 for (const auto &opt : l.attribs())
635 {
636 if (opt.name=="type")
637 {
638 m_listItemInfo[indent].type = opt.value[0];
639 }
640 if (opt.name=="start")
641 {
642 bool ok = false;
643 int val = opt.value.toInt(&ok);
644 if (ok) m_listItemInfo[indent].number = val;
645 }
646 }
647 visitChildren(l);
648 m_indent--;
649 if (!m_firstCol) m_t << "\n";
650 m_t << ".PP\n";
651}

653void ManDocVisitor::operator()(const DocHtmlListItem &li)
654{
655 if (m_hide) return;
656 QCString ws;
657 if (m_indent>0) ws.fill(' ',2*(m_indent-1));
658 if (!m_firstCol) m_t << "\n";
659 m_t << ".IP \"" << ws;
660 const DocHtmlList *list = std::get_if<DocHtmlList>(li.parent());
661 if (list && list->type()==DocHtmlList::Ordered)
662 {
663 int indent = std::min(m_indent,maxIndentLevels-1);
664 for (const auto &opt : li.attribs())
665 {
666 if (opt.name=="value")
667 {
668 bool ok = false;
669 int val = opt.value.toInt(&ok);
670 if (ok) m_listItemInfo[indent].number = val;
671 }
672 }
673 switch (m_listItemInfo[indent].type)
674 {
675 case '1':
676 m_t << m_listItemInfo[indent].number;
677 break;
678 case 'a':
679 m_t << integerToAlpha(m_listItemInfo[indent].number,false);
680 break;
681 case 'A':
682 m_t << integerToAlpha(m_listItemInfo[indent].number);
683 break;
684 case 'i':
685 m_t << integerToRoman(m_listItemInfo[indent].number,false);
686 break;
687 case 'I':
688 m_t << integerToRoman(m_listItemInfo[indent].number);
689 break;
690 default:
691 m_t << m_listItemInfo[indent].number;
692 break;
693 }
694 m_t << ".\" " << ((m_indent+1)*2);
695 m_listItemInfo[indent].number++;
696 }
697 else // bullet list
698 {
699 m_t << "\\(bu\" " << (m_indent*2);
700 }
701 m_t << "\n";
702 m_firstCol=TRUE;
703 visitChildren(li);
704 m_t << "\n";
705 m_firstCol=TRUE;
706}
