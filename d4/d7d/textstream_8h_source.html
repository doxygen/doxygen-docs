29/** @brief Text streaming class that buffers data.
30 *
31 * Simpler version of std::ostringstream that has much better
32 * performance.
33 */
34class TextStream final
35{
36 static const int INITIAL_CAPACITY = 4096;
37 public:
38 /** Creates an empty stream object.
39 */
40 explicit TextStream(size_t capacity = INITIAL_CAPACITY)
41 {
42 m_buffer.reserve(capacity);
43 }
44 /** Create a text stream object for writing to a std::ostream.
45 * @note data is buffered until flush() is called or the object is destroyed.
46 */
47 explicit TextStream(std::ostream *s) : m_s(s)
48 {
49 m_buffer.reserve(INITIAL_CAPACITY);
50 }
51 /** Create a text stream, initializing the buffer with string \a s
52 */
53 explicit TextStream(const std::string &s) : m_buffer(s)
54 {
55 m_buffer.reserve(s.length()+INITIAL_CAPACITY);
56 }
57
58 /** Writes any data that is buffered to the attached std::ostream */
59 ~TextStream() { flush(); }
60
61 TextStream(const TextStream &) = delete;
62 TextStream &operator=(const TextStream &) = delete;
63 TextStream(TextStream &&) = default;
64 TextStream &operator=(TextStream &&) = delete;
65
66 /** Sets or changes the std::ostream to write to.
67 * @note Any data already buffered will be flushed.
68 */
69 void setStream(std::ostream *s)
70 {
71 flush();
72 m_s = s;
73 m_f = nullptr;
74 }
75
76 void setFile(FILE *f)
77 {
78 flush();
79 m_s = nullptr;
80 m_f = f;
81 }
82
83 /** Returns the attached std::ostream object.
84 * @see setStream()
85 */
86 std::ostream *stream() const
87 {
88 return m_s;
89 }
90
91 FILE *file() const
92 {
93 return m_f;
94 }

96 /** Adds a character to the stream */
97 TextStream &operator<<( char c)
98 {
99 m_buffer+=c;
100 return static_cast<TextStream&>(*this);
101 }
102 /** Adds an unsigned character to the stream */
103 TextStream &operator<<( unsigned char c)
104 {
105 m_buffer+=c;
106 return static_cast<TextStream&>(*this);
107 }
108
109 /** Adds an unsigned character string to the stream */
110 TextStream &operator<<( unsigned char *s)
111 {
112 if (s)
113 {
114 unsigned char *p = s;
115 while(*p)
116 {
117 m_buffer+=*p;
118 p++;
119 }
120 }
121 return static_cast<TextStream&>(*this);
122 }

124 /** Adds a C-style string to the stream */
125 TextStream &operator<<( const char *s)
126 {
127 if (s) m_buffer+=s;
128 return static_cast<TextStream&>(*this);
129 }

173 /** Adds a size_t integer to the stream.
174 * We use SFINAE to avoid a compiler error in case size_t already matches the 'unsigned int' overload.
175 */
176 template<typename T,

179 TextStream &operator<<( T i)
180 {

182 return static_cast<TextStream&>(*this);

199 /** Adds a array of character to the stream
200 * @param buf the character buffer
201 * @param len the number of characters in the buffer to write
202 */
203 void write(const char *buf,size_t len)
204 {
205 m_buffer.append(buf,len);
206 }
207
208 /** Flushes the buffer. If a std::ostream is attached, the buffer's
209 * contents will be written to the stream.
210 */
211 void flush()
212 {
213 if (m_s)
214 {
215 m_s->write(m_buffer.c_str(),m_buffer.length());
216 }
217 else if (m_f)
218 {
219 fwrite(m_buffer.c_str(),1,m_buffer.length(),m_f);
220 }
221 m_buffer.clear();
222 }
223
224 /** Clears any buffered data */
225 void clear()
226 {
227 m_buffer.clear();
228 }
229
230 /** Return the contents of the buffer as a std::string object */
231 std::string str() const
232 {
233 return m_buffer;
234 }
235
236 /** Sets the buffer's contents to string \a s.
237 * Any data already in the buffer will be flushed.
238 */
239 void str(const std::string &s)
240 {
241 flush();
242 m_buffer=s;
243 }
244
245 /** Sets the buffer's contents to string \a s
246 * Any data already in the buffer will be flushed.
247 */
248 void str(const char *s)
249 {
250 flush();
251 if (s) m_buffer=s;
252 }
253
254 /** Returns true iff the buffer is empty */
255 bool empty() const
256 {
257 return m_buffer.empty();
258 }
259
260 private:
261 /** Writes a string representation of an integer to the buffer
262 * @param n the absolute value of the integer
263 * @param neg indicates if the integer is negative
264 */
265 void output_int32( uint32_t n, bool neg )
266 {
267 char buf[20];
268 char *p = &buf[19];
269 *p = '\0';
270 if ( neg )
271 {
272 n = static_cast<uint32_t>(-static_cast<int32_t>(n));
273 }
274 do { *--p = (static_cast<char>(n%10)) + '0'; n /= 10; } while ( n );
275 if ( neg ) *--p = '-';
276 m_buffer+=p;
277 }
278 void output_double( double d)
279 {
280 char buf[64];
281 snprintf(buf,64,"%f",d);
282 m_buffer+=buf;
283 }
284 std::string m_buffer;
285 std::ostream *m_s = nullptr;
286 FILE *m_f = nullptr;
287};
